2. Fonction à ajouter
Ajouter cette nouvelle fonction :

powershell
function Get-DomainDN {
    try {
        $domain = Get-ADDomain -ErrorAction Stop
        return $domain.DistinguishedName
    } catch {
        Write-Log "Erreur lors de la récupération du domaine: $($_.Exception.Message)" -Level Error
        throw "Impossible de récupérer les informations du domaine"
    }
}

4. Test-ADConnection à simplifier
Remplacer par :

powershell
function Test-ADConnection {
    param([int]$MaxRetries = $script:Config.MaxRetries)
    
    for ($i = 1; $i -le $MaxRetries; $i++) {
        try {
            $domain = Get-ADDomain -ErrorAction Stop
            Write-Log "Connexion AD réussie au domaine: $($domain.DNSRoot)" -Level Info
            return $true
        } catch {
            Write-Log "Tentative $i/$MaxRetries - Erreur de connexion AD: $($_.Exception.Message)" -Level Warning
            if ($i -lt $MaxRetries) {
                Start-Sleep -Seconds 2
            }
        }
    }
    return $false
}
