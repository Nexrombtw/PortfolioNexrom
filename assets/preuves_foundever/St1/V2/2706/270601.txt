# FONCTION PRINCIPALE : Initialisation
function Initialize-Application {
    try {
        Write-Log "=== DÉMARRAGE DE L'APPLICATION ===" -Level Info
        Write-Log "Version PowerShell: $($PSVersionTable.PSVersion)" -Level Info
        Write-Log "Utilisateur: $env:USERNAME" -Level Info
        Write-Log "Ordinateur: $env:COMPUTERNAME" -Level Info
        
        #  Test de connexion avec initialisation automatique du domaine
        if (-not (Test-ADConnection)) {
            Show-ErrorMessage "Connexion AD impossible" "Impossible de se connecter à Active Directory. Vérifiez votre connexion réseau et vos permissions."
            return $false
        }
        
        if (-not (Initialize-MainForm)) { return $false }
        if (-not (Initialize-SearchControls)) { return $false }
        if (-not (Initialize-UserList)) { return $false }
        if (-not (Initialize-ComputersPanel)) { return $false }  
        if (-not (Initialize-DetailsPanel)) { return $false }
        if (-not (Initialize-GroupsPanel)) { return $false }
        
        if (-not (Load-Cities)) { return $false }
        
        Write-Log "Application initialisée avec succès (avec détection automatique du domaine)" -Level Info
        return $true
        
    } catch {
        Write-Log "Erreur critique lors de l'initialisation: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur d'initialisation" "Impossible d'initialiser l'application: $($_.Exception.Message)"
        return $false
    }
}

# Système de logging avancé
function Write-Log {
    param(
        [string]$Message,
        [ValidateSet('Info', 'Warning', 'Error', 'Debug')]
        [string]$Level = 'Info'
    )
    
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $logEntry = "[$timestamp] [$Level] $Message"
    
    try {
        Add-Content -Path $script:Config.LogPath -Value $logEntry -ErrorAction SilentlyContinue
    } catch {
        # Fallback si impossible d'écrire dans le log
    }
    
    # Affichage console selon le niveau
    switch ($Level) {
        'Error' { Write-Host $logEntry -ForegroundColor Red }
        'Warning' { Write-Host $logEntry -ForegroundColor Yellow }
        'Info' { Write-Host $logEntry -ForegroundColor Green }
        'Debug' { Write-Host $logEntry -ForegroundColor Cyan }
    }
}

# Gestion des paramètres avec validation
function Load-FavoriteCity {
    try {
        if (Test-Path $script:Config.SettingsPath) {
            $json = Get-Content $script:Config.SettingsPath -Raw -ErrorAction Stop | ConvertFrom-Json -ErrorAction Stop
            if ($json.PSObject.Properties.Name -contains 'FavoriteCity') {
                Write-Log "Ville favorite chargée: $($json.FavoriteCity)" -Level Debug
                return $json.FavoriteCity
            }
        }
    } catch {
        Write-Log "Erreur lors du chargement de la ville favorite: $($_.Exception.Message)" -Level Warning
    }
    return $null
}

function Save-FavoriteCity {
    param([string]$City)
    
    $cleanCity = $City -replace "⭐ ", ""
    
    if ([string]::IsNullOrWhiteSpace($cleanCity)) {
        Write-Log "Tentative de sauvegarde d'une ville vide" -Level Warning
        return
    }
    
    try {
        $settings = @{ FavoriteCity = $cleanCity }
        $settings | ConvertTo-Json | Set-Content $script:Config.SettingsPath -ErrorAction Stop
        Write-Log "Ville favorite sauvegardée: $cleanCity" -Level Info
        
        $currentSelection = $script:UIControls.cbCity.SelectedItem
        Load-Cities
        
        $favoriteWithStar = "⭐ $cleanCity"
        if ($script:UIControls.cbCity.Items.Contains($favoriteWithStar)) {
            $script:UIControls.cbCity.SelectedItem = $favoriteWithStar
        }
        
    } catch {
        Write-Log "Erreur lors de la sauvegarde de la ville favorite: $($_.Exception.Message)" -Level Error
    }
}

# Fonction AD simplifiée
function Test-ADConnection {
    try {
        $domain = Get-ADDomain -ErrorAction Stop
        Write-Log "Connexion AD réussie au domaine: $($domain.DNSRoot)" -Level Info
        return $true
    } catch {
        Write-Log "Erreur de connexion AD: $($_.Exception.Message)" -Level Error
        return $false
    }
}

# DomainDN simple 
function Get-DomainDN {
    try {
        $domain = Get-ADDomain -ErrorAction Stop
        return $domain.DistinguishedName
    } catch {
        Write-Log "Erreur lors de la récupération du DN du domaine: $($_.Exception.Message)" -Level Error
        throw "Impossible de récupérer le DN du domaine"
    }
}

# Génération sécurisée de mot de passe
function New-SecurePassword {
    param([string]$EmployeeID)
    
    if ([string]::IsNullOrWhiteSpace($EmployeeID)) {
        throw "ID employé requis pour générer le mot de passe"
    }
    
    $randomNumbers = -join ((0..9) | Get-Random -Count 6)
    $specialChars = @('!', '@', '#', '$', '%', '&', '*')
    $specialChar = $specialChars | Get-Random
    
    return "Foundever@$EmployeeID$specialChar$randomNumbers"
}

# Fonction pour gérer les valeurs nulles
function Get-ValueOrDefault {
    param(
        [Parameter(ValueFromPipeline)]
        $Value,
        [string]$Default = ""
    )
    
    if ($null -eq $Value -or [string]::IsNullOrWhiteSpace($Value)) {
        return $Default
    }
    return $Value.ToString()
}
# FONCTION  Clear-UserDetails
function Clear-UserDetails {                                
    try {
        # Effacer tous les champs de détails
        foreach ($key in $script:UIControls.detail.Keys) {
            if ($script:UIControls.detail[$key]) {
                if ($script:UIControls.detail[$key] -is [System.Windows.Forms.TextBox]) {
                    $script:UIControls.detail[$key].Text = ""
                } elseif ($script:UIControls.detail[$key] -is [System.Windows.Forms.Label]) {
                    $script:UIControls.detail[$key].Text = ""
                    $script:UIControls.detail[$key].ForeColor = $script:Theme.Text
                }
            }
        }
        
        # Effacer la liste des groupes
        if ($script:UIControls.lbGrps) {
            $script:UIControls.lbGrps.Items.Clear()
        }
        
        # Réinitialiser les boutons
        if ($script:UIControls.btnDisable) {
            $script:UIControls.btnDisable.Text = "Désactiver compte"
            $script:UIControls.btnDisable.BackColor = $script:Theme.ButtonSecondary
        }
        
        if ($script:UIControls.btnReset) {
            $script:UIControls.btnReset.Enabled = $true
        }
        
        if ($script:UIControls.btnMoveUser) {
            $script:UIControls.btnMoveUser.Enabled = $true
        }
        
        Write-Log "Détails utilisateur effacés" -Level Debug
        
    } catch {
        Write-Log "Erreur lors de l'effacement des détails: $($_.Exception.Message)" -Level Warning
    }
}

# Chargement des villes avec étoile pour les favoris
function Load-Cities {
    try {
        $script:UIControls.cbCity.Items.Clear()
        $favoriteCity = Load-FavoriteCity
        
        # Utilisation directe d'OU=FR + domaine
        $domainDN = Get-DomainDN
        $frOU = "OU=FR,$domainDN"
        
        $root = New-Object System.DirectoryServices.DirectoryEntry("LDAP://$frOU")
        $searcher = New-Object System.DirectoryServices.DirectorySearcher($root)
        $searcher.SearchScope = [System.DirectoryServices.SearchScope]::OneLevel
        $searcher.Filter = "(objectCategory=organizationalUnit)"
        $searcher.PropertiesToLoad.Add("ou") | Out-Null
        $searcher.ServerTimeLimit = [TimeSpan]::FromSeconds($script:Config.TimeoutSeconds)
        
        $ous = $searcher.FindAll() | ForEach-Object {
            if ($_.Properties["ou"].Count -gt 0) { 
                $_.Properties["ou"][0] 
            }
        } | Sort-Object
        
        if ($ous.Count -gt 0) {
            foreach ($ou in $ous) { 
                if ($ou -eq $favoriteCity) {
                    $script:UIControls.cbCity.Items.Add("⭐ $ou")
                } else {
                    $script:UIControls.cbCity.Items.Add($ou)
                }
            }
            Write-Log "Chargé $($ous.Count) villes depuis OU=FR" -Level Info
        } else {
            Write-Log "Aucune OU trouvée dans OU=FR" -Level Warning
            return $false
        }
        
        if ($favoriteCity) {
            $favoriteWithStar = "⭐ $favoriteCity"
            if ($script:UIControls.cbCity.Items.Contains($favoriteWithStar)) {
                $script:UIControls.cbCity.SelectedItem = $favoriteWithStar
            } elseif ($script:UIControls.cbCity.Items.Contains($favoriteCity)) {
                $script:UIControls.cbCity.SelectedItem = $favoriteCity
            }
        } elseif ($script:UIControls.cbCity.Items.Count -gt 0) {
            $script:UIControls.cbCity.SelectedIndex = 0
        }
        
        return $true
    } catch {
        Write-Log "Erreur critique lors du chargement des villes: $($_.Exception.Message)" -Level Error
        return $false
    }
}

# Extraction de la ville depuis le Distinguished Name
function Get-UserCityFromDN {
    param([string]$DistinguishedName)
    
    try {
        $domainDN = Get-DomainDN
        # Pattern fixe pour OU=VilleName,OU=FR,DC=...
        if ($DistinguishedName -match "OU=([^,]+),OU=FR,$([regex]::Escape($domainDN))") {
            $extractedCity = $matches[1]
            Write-Log "Ville détectée depuis DN : $extractedCity" -Level Debug
            return $extractedCity
        }
        return $null
    } catch {
        Write-Log "Erreur lors de l'extraction de la ville : $($_.Exception.Message)" -Level Warning
        return $null
    }
}

# Synchronisation de la ville dans le ComboBox principal
function Sync-MainCitySelection {
    param([string]$CityName)
    
    try {
        if ([string]::IsNullOrWhiteSpace($CityName)) {
            return
        }
        
        # Chercher la ville avec ou sans étoile dans le ComboBox principal
        $cityWithStar = "⭐ $CityName"
        $cityFound = $false
        
        # Vérifier d'abord avec l'étoile (ville favorite)
        if ($script:UIControls.cbCity.Items.Contains($cityWithStar)) {
            $script:UIControls.cbCity.SelectedItem = $cityWithStar
            $cityFound = $true
            Write-Log "Ville favorite sélectionnée automatiquement : $CityName" -Level Info
        }
        # Sinon chercher sans étoile
        elseif ($script:UIControls.cbCity.Items.Contains($CityName)) {
            $script:UIControls.cbCity.SelectedItem = $CityName
            $cityFound = $true
            Write-Log "Ville sélectionnée automatiquement : $CityName" -Level Info
        }
        
        if ($cityFound) {
            # IMPORTANT : Déclencher manuellement l'événement de changement de ville
            # pour mettre à jour les ordinateurs et vider la liste des utilisateurs
            try {
                $script:UIControls.lbUsers.Items.Clear()
                Clear-UserDetails
                Search-ComputersUnified -CityName $CityName -SearchType "noserial" -ResultsControl $script:UIControls.lbComputers -MaxResults 100
                Write-Log "Interface mise à jour automatiquement pour la ville : $CityName" -Level Info
            } catch {
                Write-Log "Erreur lors de la mise à jour de l'interface : $($_.Exception.Message)" -Level Warning
            }
        } else {
            Write-Log "Ville '$CityName' non trouvée dans le ComboBox principal" -Level Warning
        }
        
    } catch {
        Write-Log "Erreur lors de la synchronisation de la ville : $($_.Exception.Message)" -Level Warning
    }
}

# Recherche avec affichage "Nom Prénom"
function Refresh-Users {
    try {
        $script:UIControls.lbUsers.Items.Clear()
        Clear-UserDetails
        
        if (-not $script:UIControls.cbCity.SelectedItem) {
            Write-Log "Aucune ville sélectionnée" -Level Warning
            return
        }
        
        $selectedCityRaw = $script:UIControls.cbCity.SelectedItem.ToString()
        $selectedCity = $selectedCityRaw -replace "⭐ ", ""
        
        $searchTerm = $script:UIControls.txtSearch.Text.Trim()
        
        if ([string]::IsNullOrWhiteSpace($searchTerm) -or $searchTerm -eq "Saisir un terme de recherche...") {
            Write-Log "Aucun terme de recherche - liste vide maintenue" -Level Info
            return
        }
        
        if ($searchTerm.Length -lt 2) {
            Show-InfoMessage "Veuillez saisir au moins 2 caractères pour effectuer une recherche."
            return
        }
        
        # Chemin fixe avec OU=FR
        $domainDN = Get-DomainDN
        $searchBase = "OU=$selectedCity,OU=FR,$domainDN"
        
        try {
            $null = Get-ADOrganizationalUnit -Identity $searchBase -ErrorAction Stop
        } catch {
            Show-ErrorMessage "OU introuvable" "L'unité organisationnelle '$searchBase' n'existe pas."
            return
        }
        
        Write-Log "Recherche dans $searchBase avec terme: '$searchTerm'" -Level Debug
        
        $users = Get-ADUser -SearchBase $searchBase -Filter "GivenName -like '*$searchTerm*' -or Surname -like '*$searchTerm*' -or SamAccountName -like '*$searchTerm*' -or DisplayName -like '*$searchTerm*'" -Properties Name, SamAccountName, Enabled, Department, GivenName, Surname, DisplayName |
                 Sort-Object Surname, GivenName |
                 Select-Object -First 500
        
        foreach ($user in $users) {
            $displayName = "$($user.Surname) $($user.GivenName) [$($user.SamAccountName)]"
            
            if (-not $user.Enabled) {
                $displayName += " [DÉSACTIVÉ]"
            }
            
            if ($user.Department) {
                $displayName += " - $($user.Department)"
            }
            $script:UIControls.lbUsers.Items.Add($displayName)
        }
        
        Write-Log "Trouvé $($users.Count) utilisateurs correspondant à '$searchTerm'" -Level Info
        
        if ($users.Count -eq 0) {
            Show-InfoMessage "Aucun utilisateur trouvé avec le terme de recherche '$searchTerm'."
        } elseif ($users.Count -eq 500) {
            Show-InfoMessage "Plus de 500 résultats trouvés. Seuls les 500 premiers sont affichés. Affinez votre recherche."
        }
        
    } catch {
        Write-Log "Erreur lors de la recherche d'utilisateurs: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de recherche" "Impossible de rechercher les utilisateurs: $($_.Exception.Message)"
    }
}

# Affichage des détails utilisateur
function Show-UserDetails {
    try {
        Clear-UserDetails
        
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Write-Log "Impossible d'extraire le nom d'utilisateur de: $selectedText" -Level Warning
            return
        }
        
        Write-Log "Affichage des détails pour: $samAccountName" -Level Debug
        
        $user = Get-ADUser -Identity $samAccountName -Properties * -ErrorAction Stop
        
        # Changement du texte du bouton selon l'état du compte
        if ($script:UIControls.btnDisable) {
            if ($user.Enabled) {
                $script:UIControls.btnDisable.Text = "Désactiver compte"
                $script:UIControls.btnDisable.BackColor = $script:Theme.ButtonSecondary
            } else {
                $script:UIControls.btnDisable.Text = "Activer compte"
                $script:UIControls.btnDisable.BackColor = [System.Drawing.Color]::FromArgb(34, 139, 34)
            }
        }
        
        # Remplissage des champs
        if ($script:UIControls.detail["Path"]) { 
            $script:UIControls.detail["Path"].Text = $user.DistinguishedName
        }
        if ($script:UIControls.detail["Nom"]) { 
            $script:UIControls.detail["Nom"].Text = Get-ValueOrDefault $user.Surname ""
        }
        if ($script:UIControls.detail["Prenom"]) { 
            $script:UIControls.detail["Prenom"].Text = Get-ValueOrDefault $user.GivenName ""
        }
        if ($script:UIControls.detail["Service"]) { 
            $script:UIControls.detail["Service"].Text = Get-ValueOrDefault $user.Department ""
        }
        if ($script:UIControls.detail["Matricule"]) { 
            $script:UIControls.detail["Matricule"].Text = Get-ValueOrDefault $user.EmployeeID ""
        }
        if ($script:UIControls.detail["Mail"]) { 
            $script:UIControls.detail["Mail"].Text = Get-ValueOrDefault $user.EmailAddress ""
        }
        if ($script:UIControls.detail["Session"]) { 
            $script:UIControls.detail["Session"].Text = $user.SamAccountName 
        }
        
        # État du compte
        $accountStatus = if ($user.Enabled) {
            "Compte actif"
        } else {
            "Compte désactivé"
        }
        if ($script:UIControls.detail["Etat"]) { 
            $script:UIControls.detail["Etat"].Text = $accountStatus 
        }
        
        # Dates avec formatage
        if ($script:UIControls.detail["Exp"]) {
            $script:UIControls.detail["Exp"].Text = if ($user.AccountExpirationDate) {
                $user.AccountExpirationDate.ToString("dd/MM/yyyy HH:mm")
            } else {
                "Jamais"
            }
        }
        
        if ($script:UIControls.detail["ExpPwd"]) {
            if ($user.PasswordLastSet) {
                try {
                    $pwdPolicy = Get-ADDefaultDomainPasswordPolicy
                    $pwdExpiry = $user.PasswordLastSet.AddDays($pwdPolicy.MaxPasswordAge.Days)
                    $script:UIControls.detail["ExpPwd"].Text = $pwdExpiry.ToString("dd/MM/yyyy HH:mm")
                } catch {
                    $script:UIControls.detail["ExpPwd"].Text = "Inconnu"
                }
            } else {
                $script:UIControls.detail["ExpPwd"].Text = "Jamais défini"
            }
        }
        
        if ($script:UIControls.detail["LastLogon"]) {
            $script:UIControls.detail["LastLogon"].Text = if ($user.LastLogonDate) {
                $user.LastLogonDate.ToString("dd/MM/yyyy HH:mm")
            } else {
                "Jamais connecté"
            }
        }
        
        if ($script:UIControls.detail["Created"]) {
            $script:UIControls.detail["Created"].Text = if ($user.Created) {
                $user.Created.ToString("dd/MM/yyyy HH:mm")
            } else {
                "Inconnu"
            }
        }
        
        # Chargement des groupes
        try {
            $groups = Get-ADPrincipalGroupMembership -Identity $samAccountName -ErrorAction Stop | 
                     Select-Object -ExpandProperty Name | 
                     Sort-Object
            
            if ($script:UIControls.lbGrps) {
                foreach ($group in $groups) {
                    $script:UIControls.lbGrps.Items.Add($group)
                }
            }
            
            Write-Log "Chargé $($groups.Count) groupes pour $samAccountName" -Level Debug
        } catch {
            Write-Log "Erreur lors du chargement des groupes pour $samAccountName : $($_.Exception.Message)" -Level Warning
        }
        
    } catch {
        Write-Log "Erreur lors de l'affichage des détails : $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur d'affichage" "Impossible d'afficher les détails de l'utilisateur: $($_.Exception.Message)"
    }
}

# Réinitialisation avec presse-papier et choix "changement à l'ouverture de session"
function Reset-UserPassword {
    try {
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un utilisateur."
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Show-ErrorMessage "Erreur de sélection" "Impossible d'identifier l'utilisateur sélectionné."
            return
        }
        
        $user = Get-ADUser -Identity $samAccountName -Properties EmployeeID, Enabled, Surname, GivenName -ErrorAction Stop
        
        if ([string]::IsNullOrWhiteSpace($user.EmployeeID)) {
            Show-ErrorMessage "Matricule manquant" "Le matricule est requis pour réinitialiser le mot de passe de $samAccountName."
            return
        }
        
        # Dialogue pour choisir le changement à l'ouverture de session
        $changeAtLogonForm = New-Object System.Windows.Forms.Form -Property @{
            Text = "Options de réinitialisation"
            Size = [System.Drawing.Size]::new(450, 250)
            StartPosition = "CenterParent"
            FormBorderStyle = 'FixedDialog'
            MaximizeBox = $false
            MinimizeBox = $false
            BackColor = $script:Theme.Background
        }
        
        $lblQuestion = New-Object System.Windows.Forms.Label -Property @{
            Text = "Forcer le changement de mot de passe à l'ouverture de session ?"
            Location = [System.Drawing.Point]::new(20, 20)
            Size = [System.Drawing.Size]::new(400, 40)
            Font = [System.Drawing.Font]::new("Segoe UI", 10, [System.Drawing.FontStyle]::Bold)
            ForeColor = $script:Theme.Primary
            TextAlign = 'MiddleCenter'
        }
        
        $lblUser = New-Object System.Windows.Forms.Label -Property @{
            Text = "Utilisateur : $($user.GivenName) $($user.Surname) [$samAccountName]"
            Location = [System.Drawing.Point]::new(20, 70)
            Size = [System.Drawing.Size]::new(400, 20)
            Font = [System.Drawing.Font]::new("Segoe UI", 9)
            ForeColor = $script:Theme.Text
            TextAlign = 'MiddleCenter'
        }
        
        $chkChangeAtLogon = New-Object System.Windows.Forms.CheckBox -Property @{
            Text = "L'utilisateur doit changer le mot de passe à l'ouverture de session"
            Location = [System.Drawing.Point]::new(20, 110)
            Size = [System.Drawing.Size]::new(400, 30)
            Font = [System.Drawing.Font]::new("Segoe UI", 9)
            ForeColor = $script:Theme.Text
            Checked = $true
        }
        
        $btnOK = New-Object System.Windows.Forms.Button -Property @{
            Text = "Réinitialiser"
            Location = [System.Drawing.Point]::new(250, 160)
            Size = [System.Drawing.Size]::new(100, 30)
            BackColor = $script:Theme.ButtonPrimary
            ForeColor = $script:Theme.White
            FlatStyle = 'Flat'
            DialogResult = [System.Windows.Forms.DialogResult]::OK
        }
        
        $btnCancel = New-Object System.Windows.Forms.Button -Property @{
            Text = "Annuler"
            Location = [System.Drawing.Point]::new(360, 160)
            Size = [System.Drawing.Size]::new(70, 30)
            BackColor = $script:Theme.ButtonSecondary
            ForeColor = $script:Theme.White
            FlatStyle = 'Flat'
            DialogResult = [System.Windows.Forms.DialogResult]::Cancel
        }
        
        $changeAtLogonForm.Controls.AddRange(@($lblQuestion, $lblUser, $chkChangeAtLogon, $btnOK, $btnCancel))
        
        $result = $changeAtLogonForm.ShowDialog()
        
        if ($result -ne [System.Windows.Forms.DialogResult]::OK) {
            return
        }
        
        $forceChangeAtLogon = $chkChangeAtLogon.Checked
        
        if (-not $user.Enabled) {
            Set-ADUser -Identity $samAccountName -Enabled $true
            Write-Log "Compte $samAccountName activé" -Level Info
        }
        
        $newPassword = New-SecurePassword $user.EmployeeID
        $securePassword = ConvertTo-SecureString $newPassword -AsPlainText -Force
        
        Set-ADAccountPassword -Identity $samAccountName -Reset -NewPassword $securePassword
        
        # Appliquer ou non le changement à l'ouverture de session
        if ($forceChangeAtLogon) {
            Set-ADUser -Identity $samAccountName -ChangePasswordAtLogon $true
            Write-Log "Changement de mot de passe à l'ouverture de session activé pour $samAccountName" -Level Info
            $Changenextopensession = "Il sera a modifié à l'ouverture de session."
        } else {
            Set-ADUser -Identity $samAccountName -ChangePasswordAtLogon $false
            Write-Log "Changement de mot de passe à l'ouverture de session désactivé pour $samAccountName" -Level Info
        }
        
        # NOUVEAU : Copie automatique dans le presse-papier
        try {
            $successMessage = "Bonjour,`n" +
                                "Le mot de passe de $($user.Surname) $($user.GivenName) est réinitialisé : $newPassword`n" +
                                $Changenextopensession
            [System.Windows.Forms.Clipboard]::SetText($successMessage)
            $clipboardStatus = " Mot de passe copié dans le presse-papier"
            Write-Log "Mot de passe copié dans le presse-papier pour $samAccountName" -Level Info
        } catch {
            $clipboardStatus = " Impossible de copier dans le presse-papier"
            Write-Log "Erreur lors de la copie dans le presse-papier: $($_.Exception.Message)" -Level Warning
        }
        
        # Message personnalisé avec statut du presse-papier
        $changeMessage = if ($forceChangeAtLogon) {
            "Il sera à modifier à l'ouverture de session."
        }
        
        $successMessage = "Bonjour,`n" +
                         "Le mot de passe de $($user.Surname) $($user.GivenName) est réinitialisé : $newPassword`n" +
                         "$changeMessage`n`n" +
                         "$clipboardStatus"
        
        Show-InfoMessage $successMessage "Réinitialisation réussie"
        
        Write-Log "Mot de passe réinitialisé pour $samAccountName (changement à l'ouverture: $forceChangeAtLogon)" -Level Info
        Show-UserDetails
        
    } catch {
        Write-Log "Erreur lors de la réinitialisation du mot de passe: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de réinitialisation" "Impossible de réinitialiser le mot de passe: $($_.Exception.Message)"
    }
}

# Fonction modifiée PC = Recherche avec sérial 
function Search-ComputersUnified {
    param(
        [string]$CityName,
        [string]$SearchTerm = "",
        [System.Windows.Forms.Control]$ResultsControl,
        [System.Windows.Forms.Label]$CountLabel = $null,
        [string]$SearchType = "all",
        [int]$MaxResults = 100
    )
    
    try {
        # Nettoyage des résultats
        if ($ResultsControl -is [System.Windows.Forms.ListBox]) {
            $ResultsControl.Items.Clear()
        } elseif ($ResultsControl -is [System.Windows.Forms.ListView]) {
            $ResultsControl.Items.Clear()
        }
        
        if (-not $CityName) {
            if ($CountLabel) { $CountLabel.Text = "Aucune ville sélectionnée" }
            return
        }
        
        $domainDN = Get-DomainDN  # CORRECTION DE LA CASSE
        $searchBase = "OU=$CityName,OU=FR,$domainDN"
        
        # Vérification de l'OU
        try {
            $null = Get-ADOrganizationalUnit -Identity $searchBase -ErrorAction Stop
        } catch {
            if ($CountLabel) { $CountLabel.Text = "Ville introuvable : $CityName" }
            if ($ResultsControl -is [System.Windows.Forms.ListBox]) {
                $ResultsControl.Items.Add("Ville introuvable : $CityName")
            }
            return
        }
        
        # Construction du filtre
        $filter = if ([string]::IsNullOrWhiteSpace($SearchTerm)) {
            "Enabled -eq 'True'"
        } else {
            "Enabled -eq 'True' -and Name -like '*$SearchTerm*'"
        }
        
        # Récupération des ordinateurs
        $computers = Get-ADComputer -SearchBase $searchBase -Filter $filter -Properties Name, SerialNumber, Description, OperatingSystem, LastLogonDate |
                    Sort-Object Name |
                    Select-Object -First $MaxResults
        
        # Filtrage selon le type de recherche
        switch ($SearchType) {
            "noserial" {
                $computers = $computers | Where-Object { 
                    [string]::IsNullOrWhiteSpace($_.SerialNumber) -or 
                    $_.SerialNumber -eq "To be filled by O.E.M." -or 
                    $_.SerialNumber -eq "Default string" -or 
                    $_.SerialNumber -eq "System Serial Number" -or
                    $_.SerialNumber -eq "Not Specified" -or
                    $_.SerialNumber -eq "0" -or
                    $_.SerialNumber.Length -lt 3
                }
            }
            "recent" {
                $computers = $computers | Where-Object { 
                    $_.LastLogonDate -and $_.LastLogonDate -gt (Get-Date).AddDays(-7) 
                }
            }
        }
        
        # Affichage selon le type de contrôle
        if ($ResultsControl -is [System.Windows.Forms.ListBox]) {
            # Format pour ListBox (panneau latéral)
            if ($computers.Count -eq 0) {
                $ResultsControl.Items.Add("Aucun ordinateur trouvé")
            } else {
                foreach ($computer in $computers) {
                    $displayName = $computer.Name
                    
                    # Ajout de l'OS si disponible
                    if (-not [string]::IsNullOrWhiteSpace($computer.OperatingSystem)) {
                        $osShort = $computer.OperatingSystem -replace "Microsoft Windows ", "Win"
                        $displayName += " [$osShort]"
                    }
                    
                    # Statut de connexion
                    if ($computer.LastLogonDate -and $computer.LastLogonDate -gt (Get-Date).AddDays(-7)) {
                        $displayName += " [RÉCENT]"
                    } elseif ($computer.LastLogonDate -and $computer.LastLogonDate -gt (Get-Date).AddDays(-30)) {
                        $displayName += " [ACTIF]"
                    } else {
                        $displayName += " [ANCIEN]"
                    }
                    
                    $ResultsControl.Items.Add($displayName)
                }
            }
        } elseif ($ResultsControl -is [System.Windows.Forms.ListView]) {
            # Format pour ListView (recherche détaillée)
            foreach ($computer in $computers) {
                $item = New-Object System.Windows.Forms.ListViewItem($computer.Name)
                
                $serialNumber = if ([string]::IsNullOrWhiteSpace($computer.SerialNumber) -or 
                                  $computer.SerialNumber -eq "To be filled by O.E.M.") {
                    "MANQUANT"
                } else {
                    $computer.SerialNumber
                }
                
                $lastLogon = if ($computer.LastLogonDate) {
                    $computer.LastLogonDate.ToString("dd/MM/yyyy")
                } else {
                    "Jamais"
                }
                
                $status = if ($computer.LastLogonDate -and $computer.LastLogonDate -gt (Get-Date).AddDays(-7)) {
                    "RÉCENT"
                } elseif ($computer.LastLogonDate -and $computer.LastLogonDate -gt (Get-Date).AddDays(-30)) {
                    "ACTIF"
                } else {
                    "ANCIEN"
                }
                
                $item.SubItems.Add($serialNumber)
                $item.SubItems.Add($lastLogon)
                $item.SubItems.Add($status)
                $item.SubItems.Add($computer.Description)
                $item.SubItems.Add($computer.DistinguishedName)
                
                $ResultsControl.Items.Add($item)
            }
        }
        
        # Mise à jour du compteur
        if ($CountLabel) {
            $CountLabel.Text = "$($computers.Count) ordinateur(s) trouvé(s)"
        }
        
        # Mise à jour du titre du groupe pour ListBox
        if ($ResultsControl -is [System.Windows.Forms.ListBox] -and $ResultsControl -eq $script:UIControls.lbComputers) {
            $count = $computers.Count
            if ($computers.Count -eq 0) {
                $script:UIControls.grpComputers.Text = "PC sans N° série (0)"
            } else {
                $script:UIControls.grpComputers.Text = "PC sans N° série ($count)"
            }
        }
        
        Write-Log "Recherche terminée : $($computers.Count) ordinateurs trouvés dans $CityName (type: $SearchType)" -Level Info
        
    } catch {
        Write-Log "Erreur lors de la recherche d'ordinateurs: $($_.Exception.Message)" -Level Error
        if ($CountLabel) { $CountLabel.Text = "Erreur de recherche" }
        if ($ResultsControl -is [System.Windows.Forms.ListBox]) {
            $ResultsControl.Items.Add("Erreur de chargement des ordinateurs")
        }
        Show-ErrorMessage "Erreur de recherche" "Impossible de rechercher les ordinateurs: $($_.Exception.Message)"
    }
}

# Mise à jour du numéro de série dans AD avec rafraîchissement immédiat 
function Update-ComputerSerialInAD {
    param(
        [string]$ComputerName,
        [string]$SerialNumber
    )
    
    try {
        # Vérifier que l'ordinateur existe
        $computer = Get-ADComputer -Identity $ComputerName -ErrorAction Stop
        
        $confirmMessage = "Mettre à jour le numéro de série ?`n`n" +
                         "Ordinateur : $ComputerName`n" +
                         "Nouveau numéro de série : $SerialNumber`n`n" +
                         "Cette action remplacera le numéro de série existant."
        
        if (-not (Show-ConfirmDialog $confirmMessage "Confirmation de mise à jour")) {
            return
        }
        
        # REMPLACER le numéro de série existant
        Set-ADComputer -Identity $ComputerName -Replace @{SerialNumber = $SerialNumber} -ErrorAction Stop
        
        Write-Log "Numéro de série mis à jour pour $ComputerName : $SerialNumber" -Level Info
        
        Show-InfoMessage "Numéro de série mis à jour avec succès !`n`nOrdinateur : $ComputerName`nNuméro de série : $SerialNumber" "Mise à jour réussie"
        
        # NOUVEAU : Rafraîchissement immédiat de la liste
        $selectedCityRaw = $script:UIControls.cbCity.SelectedItem.ToString()
        $selectedCity = $selectedCityRaw -replace "⭐ ", ""
        if (-not [string]::IsNullOrWhiteSpace($selectedCity)) {
            Search-ComputersUnified -CityName $selectedCity -SearchType "noserial" -ResultsControl $script:UIControls.lbComputers -MaxResults 100
        }
        
    } catch {
        Write-Log "Erreur lors de la mise à jour du numéro de série: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de mise à jour" "Impossible de mettre à jour le numéro de série dans Active Directory:`n$($_.Exception.Message)"
    }
}

# Fonction de recherche de groupes AD
function Search-ADGroups {
    try {
        # Vérification du champ de recherche
        if (-not $script:UIControls.txtGroupSearch) {
            Show-ErrorMessage "Erreur interface" "Champ de recherche des groupes introuvable."
            return
        }
        
        $searchTerm = $script:UIControls.txtGroupSearch.Text.Trim()
        
        if ([string]::IsNullOrWhiteSpace($searchTerm) -or $searchTerm -eq "Rechercher un groupe...") {
            Show-InfoMessage "Veuillez saisir un terme de recherche pour les groupes."
            return
        }
        
        if ($searchTerm.Length -lt 2) {
            Show-InfoMessage "Veuillez saisir au moins 2 caractères pour la recherche."
            return
        }
        
        Write-Log "Recherche de groupes avec terme: '$searchTerm'" -Level Debug
        
        # Recherche simplifiée et robuste
        try {
            # Test de connexion AD d'abord
            $null = Get-ADDomain -ErrorAction Stop
            
            # Recherche avec filtre simple
            $groups = Get-ADGroup -Filter "Name -like '*$searchTerm*'" -Properties Name, Description, GroupScope, GroupCategory -ErrorAction Stop |
                     Sort-Object Name |
                     Select-Object -First 100
            
            if (-not $groups -or $groups.Count -eq 0) {
                Show-InfoMessage "Aucun groupe trouvé avec le terme '$searchTerm'."
                return
            }
            
            Write-Log "Trouvé $($groups.Count) groupes" -Level Debug
            
        } catch [Microsoft.ActiveDirectory.Management.ADServerDownException] {
            Show-ErrorMessage "Serveur AD inaccessible" "Impossible de contacter le serveur Active Directory. Vérifiez votre connexion réseau."
            return
        } catch [Microsoft.ActiveDirectory.Management.ADException] {
            Show-ErrorMessage "Erreur Active Directory" "Erreur lors de la recherche dans AD :`n$($_.Exception.Message)"
            return
        } catch {
            Write-Log "Erreur lors de la recherche de groupes: $($_.Exception.Message)" -Level Error
            Show-ErrorMessage "Erreur de recherche" "Impossible de rechercher les groupes :`n$($_.Exception.Message)"
            return
        }
        
        # Interface de sélection des groupes
        $groupSelectionForm = New-Object System.Windows.Forms.Form -Property @{
            Text = "Sélectionner un groupe - $($groups.Count) résultat(s)"
            Size = [System.Drawing.Size]::new(600, 450)
            StartPosition = "CenterParent"
            FormBorderStyle = 'FixedDialog'
            MaximizeBox = $false
            MinimizeBox = $false
            BackColor = $script:Theme.Background
        }
        
        # En-tête
        $lblHeader = New-Object System.Windows.Forms.Label -Property @{
            Text = "Recherche : '$searchTerm' - $($groups.Count) groupe(s) trouvé(s)"
            Location = [System.Drawing.Point]::new(20, 15)
            Size = [System.Drawing.Size]::new(550, 25)
            Font = [System.Drawing.Font]::new("Segoe UI", 10, [System.Drawing.FontStyle]::Bold)
            ForeColor = $script:Theme.Primary
            TextAlign = 'MiddleCenter'
        }
        
        # Liste des groupes
        $lbGroups = New-Object System.Windows.Forms.ListBox -Property @{
            Location = [System.Drawing.Point]::new(20, 50)
            Size = [System.Drawing.Size]::new(550, 300)
            Font = [System.Drawing.Font]::new("Segoe UI", 9)
            BackColor = $script:Theme.White
            ForeColor = $script:Theme.Text
        }
        
        # Remplissage de la liste
        foreach ($group in $groups) {
            $displayText = $group.Name
            if ($group.Description) {
                $displayText += " - $($group.Description)"
            }
            $displayText += " [$($group.GroupCategory)/$($group.GroupScope)]"
            
            $lbGroups.Items.Add($displayText)
            # Stocker le nom du groupe dans la propriété Tag (simulation)
            $lbGroups.Tag = if ($lbGroups.Tag) { $lbGroups.Tag + ";" + $group.Name } else { $group.Name }
        }
        
        # Boutons
        $btnSelect = New-Object System.Windows.Forms.Button -Property @{
            Text = "Ajouter au groupe sélectionné"
            Location = [System.Drawing.Point]::new(350, 370)
            Size = [System.Drawing.Size]::new(150, 30)
            BackColor = $script:Theme.ButtonPrimary
            ForeColor = $script:Theme.White
            FlatStyle = 'Flat'
            Enabled = $false
        }
        
        $btnCancel = New-Object System.Windows.Forms.Button -Property @{
            Text = "Annuler"
            Location = [System.Drawing.Point]::new(510, 370)
            Size = [System.Drawing.Size]::new(60, 30)
            BackColor = $script:Theme.ButtonSecondary
            ForeColor = $script:Theme.White
            FlatStyle = 'Flat'
        }
        
        # Événements
        $lbGroups.Add_SelectedIndexChanged({
            $btnSelect.Enabled = ($lbGroups.SelectedIndex -ge 0)
        })
        
        $lbGroups.Add_DoubleClick({
            if ($lbGroups.SelectedIndex -ge 0) {
                $btnSelect.PerformClick()
            }
        })
        
        $btnSelect.Add_Click({
            if ($lbGroups.SelectedIndex -ge 0) {
                # Récupérer le nom du groupe sélectionné
                $groupNames = $lbGroups.Tag -split ";"
                $selectedGroupName = $groupNames[$lbGroups.SelectedIndex]
                
                try {
                    Add-UserToSelectedGroup $selectedGroupName
                    $groupSelectionForm.Close()
                } catch {
                    Show-ErrorMessage "Erreur lors de l'ajout au groupe" $_.Exception.Message
                }
            }
        })
        
        $btnCancel.Add_Click({
            $groupSelectionForm.Close()
        })
        
        $groupSelectionForm.Controls.AddRange(@($lblHeader, $lbGroups, $btnSelect, $btnCancel))
        $groupSelectionForm.ShowDialog()
        
    } catch {
        Write-Log "Erreur critique lors de la recherche de groupes: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur critique" "Erreur inattendue lors de la recherche de groupes :`n$($_.Exception.Message)"
    }
}

# Fonction pour ajouter un utilisateur au groupe sélectionné
function Add-UserToSelectedGroup {
    param([string]$GroupName)
    
    # Validation des paramètres
    if ([string]::IsNullOrWhiteSpace($GroupName)) {
        Show-ErrorMessage "Erreur" "Nom de groupe invalide."
        return
    }
    
    # Vérification de la sélection utilisateur
    if (-not $script:UIControls.lbUsers.SelectedItem) {
        Show-InfoMessage "Veuillez sélectionner un utilisateur."
        return
    }
    
    $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
    if ($selectedText -match '\[([^\]]+)\]') {
        $samAccountName = $matches[1]
    } else {
        Show-ErrorMessage "Erreur de sélection" "Impossible d'identifier l'utilisateur sélectionné."
        return
    }
    
    try {
        Write-Log "Tentative d'ajout de $samAccountName au groupe $GroupName" -Level Debug
        
        # Vérifications préalables
        # Vérifier que l'utilisateur existe
        try {
            $user = Get-ADUser -Identity $samAccountName -Properties Enabled -ErrorAction Stop
        } catch {
            Show-ErrorMessage "Utilisateur introuvable" "L'utilisateur '$samAccountName' n'existe pas dans Active Directory."
            return
        }
        
        # Vérifier que le groupe existe
        try {
            $group = Get-ADGroup -Identity $GroupName -Properties GroupScope, GroupCategory -ErrorAction Stop
        } catch {
            Show-ErrorMessage "Groupe introuvable" "Le groupe '$GroupName' n'existe pas dans Active Directory."
            return
        }
        
        # Vérifier l'appartenance actuelle
        try {
            $currentGroups = Get-ADPrincipalGroupMembership -Identity $samAccountName -ErrorAction Stop | 
                            Select-Object -ExpandProperty Name
            
            if ($currentGroups -contains $GroupName) {
                Show-InfoMessage "L'utilisateur '$samAccountName' est déjà membre du groupe '$GroupName'."
                return
            }
        } catch {
            Write-Log "Impossible de vérifier l'appartenance actuelle: $($_.Exception.Message)" -Level Warning
            # Continuer quand même
        }
        
        # Confirmation
        $confirmMessage = "Ajouter l'utilisateur au groupe ?`n`n" +
                         "Utilisateur : $samAccountName`n" +
                         "Groupe : $GroupName`n" +
                         "Type : $($group.GroupCategory) ($($group.GroupScope))"
        
        if (-not (Show-ConfirmDialog $confirmMessage "Confirmation d'ajout")) {
            return
        }
        
        # Ajout avec gestion d'erreurs détaillée
        try {
            Add-ADGroupMember -Identity $GroupName -Members $samAccountName -ErrorAction Stop
            Write-Log "Utilisateur $samAccountName ajouté au groupe $GroupName avec succès" -Level Info
            
            $successMessage = "AJOUT RÉUSSI`n`n" +
                             "Utilisateur : $samAccountName`n" +
                             "Groupe : $GroupName`n" +
                             "Ajout effectué le : $(Get-Date -Format "dd/MM/yyyy à HH:mm:ss")"
            
            Show-InfoMessage $successMessage "Ajout réussi"
            
            # Rafraîchir l'affichage
            Show-UserDetails
            
        } catch [Microsoft.ActiveDirectory.Management.ADException] {
            if ($_.Exception.Message -match "insufficient access rights|access.*denied") {
                Show-ErrorMessage "Permissions insuffisantes" "Vous n'avez pas les droits pour ajouter des membres à ce groupe.`n`nContactez votre administrateur système."
            } elseif ($_.Exception.Message -match "already.*member") {
                Show-InfoMessage "L'utilisateur est déjà membre de ce groupe."
                Show-UserDetails
            } else {
                Show-ErrorMessage "Erreur Active Directory" "Erreur lors de l'ajout au groupe :`n$($_.Exception.Message)"
            }
        } catch {
            Write-Log "Erreur inattendue: $($_.Exception.Message)" -Level Error
            Show-ErrorMessage "Erreur inattendue" "Une erreur s'est produite :`n$($_.Exception.Message)"
        }
        
    } catch {
        Write-Log "Erreur critique dans Add-UserToSelectedGroup: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur critique" "Erreur inattendue :`n$($_.Exception.Message)"
    }
}

# Suppression d'utilisateur d'un groupe
function Remove-UserFromGroup {
    try {
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un utilisateur."
            return
        }
        
        if (-not $script:UIControls.lbGrps.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un groupe dans la liste des appartenances."
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Show-ErrorMessage "Erreur de sélection" "Impossible d'identifier l'utilisateur sélectionné."
            return
        }
        
        $groupName = $script:UIControls.lbGrps.SelectedItem.ToString()
        
        if ($groupName -eq "Domain Users") {
            Show-ErrorMessage "Action interdite" "Impossible de retirer un utilisateur du groupe 'Domain Users'."
            return
        }
        
        if (-not (Show-ConfirmDialog "Retirer $samAccountName du groupe '$groupName' ?" "Confirmation de suppression")) {
            return
        }
        
        Remove-ADGroupMember -Identity $groupName -Members $samAccountName -Confirm:$false
        Write-Log "Utilisateur $samAccountName retiré du groupe $groupName" -Level Info
        
        Show-InfoMessage "Utilisateur retiré du groupe '$groupName' avec succès !"
        Show-UserDetails
        
    } catch {
        Write-Log "Erreur lors de la suppression du groupe: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de suppression" "Impossible de retirer l'utilisateur du groupe: $($_.Exception.Message)"
    }
}

# Déplacement d'utilisateur avec synchronisation automatique de la ville
function Move-UserToOU {
    try {
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un utilisateur."
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Show-ErrorMessage "Erreur" "Impossible d'identifier l'utilisateur sélectionné."
            return
        }
        
        # Récupération des informations utilisateur
        $user = Get-ADUser -Identity $samAccountName -Properties DistinguishedName, GivenName, Surname -ErrorAction Stop
        
        # Détection automatique de la ville de l'utilisateur
        $detectedCity = Get-UserCityFromDN $user.DistinguishedName
        
        # Synchronisation automatique de la ville dans le ComboBox principal

            Sync-MainCitySelection $detectedCity
            Write-Log "Ville principale synchronisée automatiquement : $detectedCity" -Level Info
        }
        
        # Interface de déplacement avec ville pré-sélectionnée
        Show-UniformMoveInterface $user $detectedCity
        
    } catch {
        Write-Log "Erreur lors du déplacement: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur" "Impossible d'ouvrir la fenêtre de déplacement: $($_.Exception.Message)"
    }
}

# Fonction pour désactiver un compte utilisateur
function Disable-UserAccount {
    try {
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un utilisateur."
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Show-ErrorMessage "Erreur de sélection" "Impossible d'identifier l'utilisateur sélectionné."
            return
        }
        
        $user = Get-ADUser -Identity $samAccountName -Properties Surname, GivenName, Enabled, DistinguishedName, Description -ErrorAction Stop
        
        if (-not $user.Enabled) {
            Show-InfoMessage "Le compte de $($user.GivenName) $($user.Surname) est déjà désactivé."
            return
        }
        
        # Déterminer la ville d'origine avec la nouvelle structure OU=FR
        $currentDN = $user.DistinguishedName
        $userCity = $null
        
        # Extraction de ville avec pattern OU=FR
        $extractedCity = Get-UserCityFromDN $currentDN
        if ($extractedCity) {
            $userCity = $extractedCity
            Write-Log "Ville d'origine détectée : $userCity pour $samAccountName" -Level Info
        } else {
            Write-Log "Impossible de détecter la ville d'origine pour $samAccountName" -Level Warning
        }
        
        $confirmMessage = "ATTENTION - DÉSACTIVATION DE COMPTE`n`n" +
                         "Vous êtes sur le point de désactiver le compte de :`n`n" +
                         "Utilisateur : $($user.GivenName) $($user.Surname)`n" +
                         "Identifiant : $samAccountName`n"
        
        if ($userCity) {
            $confirmMessage += "Ville d'origine détectée : $userCity`n`n"
        } else {
            $confirmMessage += "Ville d'origine : Non détectée`n`n"
        }
        
        $confirmMessage += "Cette action va :`n" +
                          "• Désactiver le compte`n"
        
        if ($userCity) {
            $confirmMessage += "• Déplacer vers l'OU 'Disable Users' de $userCity`n"
        } else {
            $confirmMessage += "• Laisser le compte dans son emplacement actuel (ville non détectée)`n"
        }
        
        $confirmMessage += "• Retirer de tous les groupes (sauf Domain Users)`n`n" +
                          "Voulez-vous continuer ?"
        
        if (-not (Show-ConfirmDialog $confirmMessage "Confirmation de désactivation")) {
            return
        }
        
        # Construire le chemin de l'OU avec structure OU=FR (seulement si ville détectée)
        $ouExists = $false
        $disabledOUPath = $null
        
        if ($userCity) {
            $domainDN = Get-DomainDN
            $disabledOUPath = "OU=Disable Users,OU=Others,OU=$userCity,OU=FR,$domainDN"
            
            try {
                $null = Get-ADOrganizationalUnit -Identity $disabledOUPath -ErrorAction Stop
                $ouExists = $true
                Write-Log "OU Disable Users trouvée pour $userCity : $disabledOUPath" -Level Info
            } catch {
                Write-Log "OU 'Disable Users' introuvable pour $userCity : $disabledOUPath" -Level Warning
                $ouExists = $false
            }
        }
        
        # Désactiver le compte
        try {
            Disable-ADAccount -Identity $samAccountName -ErrorAction Stop
            Write-Log "Compte $samAccountName désactivé avec succès" -Level Info
            
            # MISE À JOUR AUTOMATIQUE DE L'AFFICHAGE DANS LA LISTE
            $currentIndex = $script:UIControls.lbUsers.SelectedIndex
            if ($currentIndex -ge 0) {
                $displayText = $script:UIControls.lbUsers.Items[$currentIndex].ToString()
                if ($displayText -notmatch '\[DÉSACTIVÉ\]') {
                    # Ajouter le marqueur [DÉSACTIVÉ] après le nom d'utilisateur
                    $newDisplayText = $displayText -replace '(\[[^\]]+\])', '$1 [DÉSACTIVÉ]'
                    $script:UIControls.lbUsers.Items[$currentIndex] = $newDisplayText
                }
            }
            
            # MISE À JOUR AUTOMATIQUE DU BOUTON
            if ($script:UIControls.btnDisable) {
                $script:UIControls.btnDisable.Text = "Activer compte"
                $script:UIControls.btnDisable.BackColor = [System.Drawing.Color]::FromArgb(34, 139, 34)
            }
            
            # MISE À JOUR AUTOMATIQUE DE L'ÉTAT DANS LES DÉTAILS
            if ($script:UIControls.detail["Etat"]) {
                $script:UIControls.detail["Etat"].Text = "Compte désactivé"
                $script:UIControls.detail["Etat"].ForeColor = [System.Drawing.Color]::Red
            }
            
        } catch {
            Write-Log "Erreur lors de la désactivation du compte: $($_.Exception.Message)" -Level Error
            Show-ErrorMessage "Erreur de désactivation" "Impossible de désactiver le compte: $($_.Exception.Message)"
            return
        }
        
        # Déplacer vers l'OU appropriée (seulement si ville détectée et OU existe)
        $moveResult = ""
        if ($userCity -and $ouExists) {
            try {
                $originalDescription = $user.Description
                Move-ADObject -Identity $user.DistinguishedName -TargetPath $disabledOUPath -ErrorAction Stop
                Write-Log "Compte $samAccountName déplacé vers OU Disable Users de $userCity" -Level Info
                
                # Restaurer la description avec retry
                if (-not [string]::IsNullOrEmpty($originalDescription)) {
                    $retryCount = 0
                    do {
                        try {
                            Start-Sleep -Milliseconds 500
                            Set-ADUser -Identity $samAccountName -Description $originalDescription -ErrorAction Stop
                            Write-Log "Description restaurée pour $samAccountName" -Level Debug
                            break
                        } catch {
                            $retryCount++
                            if ($retryCount -ge 3) {
                                Write-Log "Impossible de restaurer la description après 3 tentatives" -Level Warning
                                break
                            }
                            Start-Sleep -Seconds 1
                        }
                    } while ($retryCount -lt 3)
                }
                
                # MISE À JOUR DU CHEMIN AD DANS LES DÉTAILS
                if ($script:UIControls.detail["Path"]) {
                    $script:UIControls.detail["Path"].Text = $disabledOUPath
                }
                
                $moveResult = "• Déplacé vers l'OU Disable Users de $userCity"
            } catch {
                Write-Log "Erreur lors du déplacement: $($_.Exception.Message)" -Level Warning
                $moveResult = "• Erreur lors du déplacement: $($_.Exception.Message)"
            }
        } elseif ($userCity -and -not $ouExists) {
            $moveResult = "• OU 'Disable Users' introuvable pour $userCity - compte non déplacé"
        } else {
            $moveResult = "• Ville non détectée - compte laissé dans son emplacement actuel"
        }
        
        # Retirer des groupes (version sécurisée)
        $groupsRemoved = 0
        $groupErrors = @()
        try {
            $userGroups = Get-ADPrincipalGroupMembership -Identity $samAccountName -ErrorAction Stop
            foreach ($group in $userGroups) {
                if ($group.Name -ne "Domain Users") {
                    try {
                        Remove-ADGroupMember -Identity $group.Name -Members $samAccountName -Confirm:$false -ErrorAction Stop
                        $groupsRemoved++
                        Write-Log "Utilisateur $samAccountName retiré du groupe $($group.Name)" -Level Debug
                    } catch {
                        $errorMsg = "Erreur suppression groupe $($group.Name): $($_.Exception.Message)"
                        $groupErrors += $errorMsg
                        Write-Log $errorMsg -Level Warning
                    }
                }
            }
            
            # MISE À JOUR DE LA LISTE DES GROUPES
            if ($script:UIControls.lbGrps) {
                $script:UIControls.lbGrps.Items.Clear()
                $script:UIControls.lbGrps.Items.Add("Domain Users")  # Seul groupe restant
            }
            
        } catch {
            Write-Log "Erreur lors de la récupération des groupes: $($_.Exception.Message)" -Level Warning
            $groupErrors += "Impossible de récupérer la liste des groupes"
        }
        
        $groupResult = if ($groupErrors.Count -gt 0) {
            "• Retiré de $groupsRemoved groupe(s) avec $($groupErrors.Count) erreur(s)"
        } else {
            "• Retiré de $groupsRemoved groupe(s)"
        }
        
        $successMessage = "DÉSACTIVATION TERMINÉE`n`n" +
                         "Le compte de $($user.GivenName) $($user.Surname) a été traité.`n`n" +
                         "Actions effectuées :`n" +
                         "• Compte désactivé`n" +
                         "$moveResult`n" +
                         "$groupResult`n`n" +
                         "Désactivation effectuée le : $(Get-Date -Format "dd/MM/yyyy à HH:mm:ss")`n" +
                         "Par : $env:USERNAME"
        
        Show-InfoMessage $successMessage "Désactivation terminée"
        
        Write-Log "Processus de désactivation terminé pour $samAccountName" -Level Info
        
    } catch {
        Write-Log "Erreur critique lors de la désactivation: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur critique" "Erreur lors de la désactivation du compte: $($_.Exception.Message)"
    }
}

# Fonction pour activer un compte utilisateur
function Enable-UserAccount {
    try {
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un utilisateur."
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Show-ErrorMessage "Erreur de sélection" "Impossible d'identifier l'utilisateur sélectionné."
            return
        }
        
        $user = Get-ADUser -Identity $samAccountName -Properties Surname, GivenName, Enabled, DistinguishedName, Description -ErrorAction Stop
        
        if ($user.Enabled) {
            Show-InfoMessage "Le compte de $($user.GivenName) $($user.Surname) est déjà activé."
            return
        }
        
        $confirmMessage = "ACTIVATION DE COMPTE`n`n" +
                         "Vous êtes sur le point d'activer le compte :`n`n" +
                         "Utilisateur : $($user.GivenName) $($user.Surname)`n" +
                         "Identifiant : $samAccountName`n`n" +
                         "Souhaitez-vous continuer ?"
        
        if (-not (Show-ConfirmDialog $confirmMessage "Confirmation d'activation")) {
            return
        }
        
        # Activer le compte
        try {
            Enable-ADAccount -Identity $samAccountName -ErrorAction Stop

            if ($currentIndex -ge 0) {
                $displayText = $script:UIControls.lbUsers.Items[$currentIndex].ToString()
                if ($displayText -match '\[DÉSACTIVÉ\]') {
                    # Enlever le marqueur [DÉSACTIVÉ] de l'affichage
                    $newDisplayText = $displayText -replace ' \[DÉSACTIVÉ\]', ''
                    $script:UIControls.lbUsers.Items[$currentIndex] = $newDisplayText
                }
            }
            
            # MISE À JOUR AUTOMATIQUE DU BOUTON
            if ($script:UIControls.btnDisable) {
                $script:UIControls.btnDisable.Text = "Désactiver compte"
                $script:UIControls.btnDisable.BackColor = $script:Theme.ButtonSecondary
            }
            
            # MISE À JOUR AUTOMATIQUE DE L'ÉTAT DANS LES DÉTAILS
            if ($script:UIControls.detail["Etat"]) {
                $script:UIControls.detail["Etat"].Text = "Compte actif"
                $script:UIControls.detail["Etat"].ForeColor = [System.Drawing.Color]::Gray  
            }
            
        } catch {
            Write-Log "Erreur lors de l'activation du compte: $($_.Exception.Message)" -Level Error
            Show-ErrorMessage "Erreur d'activation" "Impossible d'activer le compte: $($_.Exception.Message)"
            return
        }
        
        # Préserver la description
        $originalDescription = $user.Description
        if (-not [string]::IsNullOrEmpty($originalDescription)) {
            Set-ADUser -Identity $samAccountName -Description $originalDescription -ErrorAction SilentlyContinue
        }
        
        # Réinitialiser le mot de passe avec choix
        $user2 = Get-ADUser -Identity $samAccountName -Properties EmployeeID -ErrorAction Stop
        if (-not [string]::IsNullOrWhiteSpace($user2.EmployeeID)) {
            $newPassword = New-SecurePassword $user2.EmployeeID
            $securePassword = ConvertTo-SecureString $newPassword -AsPlainText -Force
            Set-ADAccountPassword -Identity $samAccountName -Reset -NewPassword $securePassword
            Set-ADUser -Identity $samAccountName -ChangePasswordAtLogon $true
            
            # Copie automatique dans le presse-papier
            try {
                [System.Windows.Forms.Clipboard]::SetText($newPassword)
                $clipboardStatus = " Mot de passe copié dans le presse-papier"
                Write-Log "Mot de passe copié dans le presse-papier pour $samAccountName" -Level Info
            } catch {
                $clipboardStatus = " Impossible de copier dans le presse-papier"
                Write-Log "Erreur lors de la copie dans le presse-papier: $($_.Exception.Message)" -Level Warning
            }
            
            $passwordMessage = "Bonjour,`nLe mot de passe de $($user.Surname) $($user.GivenName) est réinitialisé : $newPassword`nIl sera à modifier à l'ouverture de session.`n`n$clipboardStatus"
            Show-InfoMessage $passwordMessage "Mot de passe"
        }
        
        # Correspondance avec la ville sélectionnée dans la fenêtre principale
        $selectedCityRaw = $script:UIControls.cbCity.SelectedItem.ToString()
        $selectedCity = $selectedCityRaw -replace "⭐ ", ""
        
        # Interface de déplacement pré-configurée avec la ville correspondante
        $updatedUser = Get-ADUser -Identity $samAccountName -Properties DistinguishedName,GivenName,Surname -ErrorAction Stop
        Show-UniformMoveInterface $updatedUser $selectedCity
        
        Write-Log "Processus d'activation terminé pour $samAccountName" -Level Info
        Show-UserDetails
        
    } catch {
        Write-Log "Erreur critique lors de l'activation: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur critique" "Erreur lors de l'activation du compte: $($_.Exception.Message)"
    }
}

# Fonction pour charger les villes
function Load-AvailableCitiesForMove {
    param([System.Windows.Forms.ComboBox]$ComboBox)
    
    try {
        $ComboBox.Items.Clear()
        
        # Chemin fixe vers OU=FR
        $domainDN = Get-DomainDN
        $frOU = "OU=FR,$domainDN"
        
        $root = New-Object System.DirectoryServices.DirectoryEntry("LDAP://$frOU")
        $searcher = New-Object System.DirectoryServices.DirectorySearcher($root)
        $searcher.Filter = "(objectCategory=organizationalUnit)"
        $searcher.SearchScope = [System.DirectoryServices.SearchScope]::OneLevel
        $searcher.PropertiesToLoad.Add("ou") | Out-Null
        
        $cities = $searcher.FindAll() | ForEach-Object {
            if ($_.Properties["ou"].Count -gt 0) { 
                $_.Properties["ou"][0] 
            }
        } | Sort-Object | Select-Object -Unique
        
        $favoriteCity = Load-FavoriteCity
        
        if ($favoriteCity -and $cities -contains $favoriteCity) {
            $ComboBox.Items.Add("⭐ $favoriteCity")
        }
        
        foreach ($city in $cities) {
            if ($city -ne $favoriteCity) {
                $ComboBox.Items.Add($city)
            }
        }
        
    } catch {
        Write-Log "Erreur lors du chargement des villes: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur" "Impossible de charger la liste des villes: $($_.Exception.Message)"
    }
}

# FONCTION UNIQUE COMPLÈTE - Remplace Load-UniformCityFolders, Build-UniformFolderTree et Get-UniformFolderDisplayName
function Load-UniformCityFolders {
    param(
        [string]$SelectedCity,
        [System.Windows.Forms.TreeView]$TreeView,
        [System.Windows.Forms.Label]$InfoLabel,
        [string]$CurrentUserPath
    )
    
    try {
        $TreeView.BeginUpdate()
        $TreeView.Nodes.Clear()
        
        # Chemin fixe avec OU=FR
        $domainDN = Get-DomainDN
        $cityBasePath = "OU=$SelectedCity,OU=FR,$domainDN"
        
        Write-Log "Chargement des dossiers pour $SelectedCity depuis $cityBasePath" -Level Debug
        
        # Vérifier que l'OU de la ville existe
        try {
            $null = Get-ADOrganizationalUnit -Identity $cityBasePath -ErrorAction Stop
            Write-Log "OU ville confirmée : $cityBasePath" -Level Debug
        } catch {
            $InfoLabel.Text = "Ville '$SelectedCity' introuvable dans AD"
            $InfoLabel.ForeColor = [System.Drawing.Color]::Red
            Write-Log "OU ville introuvable : $cityBasePath" -Level Error
            return
        }
        
        # Dossiers à exclure pour une interface claire
        $excludedFolders = @(
            "Groups", "Functions", "Computers", "Service Accounts", 
            "Resources", "Contacts", "Distribution Lists", "Security Groups",
            "Shared Mailboxes", "Room Mailboxes", "Equipment", "Templates",
            "System", "Builtin", "ForeignSecurityPrincipals", "Managed Service Accounts"
        )
        
        # Récupération de TOUTES les OUs sous la ville (sauf exclusions)
        $allOUs = Get-ADOrganizationalUnit -SearchBase $cityBasePath -SearchScope Subtree -Filter * |
                  Where-Object { $excludedFolders -notcontains $_.Name } |
                  Sort-Object DistinguishedName
        
        Write-Log "Trouvé $($allOUs.Count) OUs sous $cityBasePath" -Level Debug
        
        if ($allOUs.Count -eq 0) {
            $noFolderNode = $TreeView.Nodes.Add("Aucun dossier disponible dans cette ville")
            $noFolderNode.Tag = $cityBasePath
            $noFolderNode.ForeColor = [System.Drawing.Color]::Gray
            $InfoLabel.Text = "Aucun sous-dossier trouvé dans $SelectedCity"
            return
        }
        
        # Ajouter d'abord le dossier racine de la ville
        $rootNode = $TreeView.Nodes.Add("📁 $SelectedCity (Racine)")
        $rootNode.Tag = $cityBasePath
        $rootNode.ToolTipText = $cityBasePath
        $rootNode.ForeColor = [System.Drawing.Color]::Black
        $rootNode.NodeFont = [System.Drawing.Font]::new("Segoe UI", 9, [System.Drawing.FontStyle]::Bold)
        
        if ($cityBasePath -eq $CurrentUserPath) {
            $rootNode.Text += " [EMPLACEMENT ACTUEL]"
            $rootNode.ForeColor = [System.Drawing.Color]::Blue
        }
        
        # Organisation hiérarchique des dossiers
        $folderDict = @{}
        foreach ($ou in $allOUs) {
            $parentDN = ($ou.DistinguishedName -split ',', 2)[1]
            if (-not $folderDict.ContainsKey($parentDN)) {
                $folderDict[$parentDN] = @()
            }
            $folderDict[$parentDN] += $ou
        }
        
        # CONSTRUCTION DE L'ARBRE INTÉGRÉE (remplace Build-UniformFolderTree)
        function Build-TreeRecursive {
            param(
                [System.Windows.Forms.TreeNode]$ParentNode,
                [string]$ParentDN,
                [hashtable]$FolderDict,
                [string]$CurrentUserPath
            )
            
            if ($FolderDict.ContainsKey($ParentDN)) {
                foreach ($ou in ($FolderDict[$ParentDN] | Sort-Object Name)) {
                    
                    # NOMS D'AFFICHAGE INTÉGRÉS (remplace Get-UniformFolderDisplayName)
                    $displayName = switch ($ou.Name) {
                        "Administrative" { "📋 Service Administratif" }
                        "Commercial" { "💼 Service Commercial" }
                        "Technique" { "🔧 Service Technique" }
                        "Direction" { "👔 Direction" }
                        "RH" { "👥 Ressources Humaines" }
                        "IT" { "💻 Informatique" }
                        "Others" { "📁 Autres utilisateurs" }
                        "Finance" { "💰 Finance" }
                        "Marketing" { "📈 Marketing" }
                        "Support" { "🎧 Support" }
                        "Disable Users" { "❌ Comptes désactivés" }
                        "Users" { "👤 Utilisateurs" }
                        default { "📁 $($ou.Name)" }
                    }
                    
                    # Création du nœud
                    $node = $ParentNode.Nodes.Add($displayName)
                    $node.Tag = $ou.DistinguishedName
                    $node.Name = $ou.Name
                    $node.ToolTipText = "Chemin : $($ou.DistinguishedName)"
                    
                    # Indication visuelle si c'est l'emplacement actuel
                    if ($ou.DistinguishedName -eq $CurrentUserPath) {
                        $node.Text += " [EMPLACEMENT ACTUEL]"
                        $node.ForeColor = [System.Drawing.Color]::Blue
                        $node.NodeFont = [System.Drawing.Font]::new("Segoe UI", 9, [System.Drawing.FontStyle]::Bold)
                    } else {
                        # COLORATION UNIFORME 
                        $node.ForeColor = [System.Drawing.Color]::Black
                        $node.NodeFont = [System.Drawing.Font]::new("Segoe UI", 9, [System.Drawing.FontStyle]::Regular)
                    }
                    
                    # Construction récursive des sous-dossiers
                    Build-TreeRecursive $node $ou.DistinguishedName $FolderDict $CurrentUserPath
                }
            }
        }
        
        # Appel de la construction récursive
        Build-TreeRecursive $rootNode $cityBasePath $folderDict $CurrentUserPath
        
        # Expansion automatique
        $rootNode.Expand()
        foreach ($node in $rootNode.Nodes) {
            if ($node.Nodes.Count -gt 0) {
                $node.Expand()
            }
        }
        
        $InfoLabel.Text = "Dossiers chargés pour $SelectedCity ($($allOUs.Count + 1) dossiers disponibles)"
        $InfoLabel.ForeColor = $script:Theme.Primary
        
    } catch {
        Write-Log "Erreur lors du chargement des dossiers: $($_.Exception.Message)" -Level Error
        $InfoLabel.Text = "Erreur lors du chargement des dossiers"
        $InfoLabel.ForeColor = [System.Drawing.Color]::Red
        Show-ErrorMessage "Erreur" "Impossible de charger les dossiers: $($_.Exception.Message)"
    } finally {
        $TreeView.EndUpdate()
    }
}

# Exécution du déplacement 
function Execute-UniformUserMove {
    param(
        [Microsoft.ActiveDirectory.Management.ADUser]$User,
        [string]$TargetPath,
        [string]$DisplayName,
        [System.Windows.Forms.Form]$ParentForm
    )
    
    try {
        $confirmMessage = "CONFIRMATION DE DÉPLACEMENT`n`n" +
                         "Utilisateur : $($User.GivenName) $($User.Surname) [$($User.SamAccountName)]`n`n" +
                         "DEPUIS :`n$($User.DistinguishedName)`n`n" +
                         "VERS :`n$TargetPath`n`n" +
                         "Dossier de destination : $DisplayName`n`n" +
                         "Cette action est irréversible. Continuer ?"
        
        if (Show-ConfirmDialog $confirmMessage "Confirmation de déplacement") {
            # Sauvegarder la description actuelle avant le déplacement
            $currentUser = Get-ADUser -Identity $User.SamAccountName -Properties Description -ErrorAction Stop
            $originalDescription = $currentUser.Description
            
            # Exécution du déplacement
            Move-ADObject -Identity $User.DistinguishedName -TargetPath $TargetPath -ErrorAction Stop
            
            # Restaurer la description originale après le déplacement
            if (-not [string]::IsNullOrEmpty($originalDescription)) {
                Set-ADUser -Identity $User.SamAccountName -Description $originalDescription -ErrorAction SilentlyContinue
                Write-Log "Description originale restaurée pour $($User.SamAccountName): $originalDescription" -Level Debug
            }
            
            Write-Log "Utilisateur $($User.SamAccountName) déplacé avec succès vers $TargetPath (description préservée)" -Level Info
            
            $successMessage = "DÉPLACEMENT RÉUSSI`n`n" +
                             "L'utilisateur $($User.GivenName) $($User.Surname) a été déplacé avec succès.`n`n" +
                             "Nouveau dossier : $DisplayName`n" +
                             "Nouveau chemin : $TargetPath`n`n" +
                             "Description AD préservée`n`n" +
                             "Déplacement effectué le : $(Get-Date -Format "dd/MM/yyyy à HH:mm:ss")`n" +
                             "Par : $env:USERNAME"
            
            Show-InfoMessage $successMessage "Déplacement réussi"
            
            # Fermeture du formulaire et rafraîchissement
            $ParentForm.Close()
            Show-UserDetails
        }
    } catch {
        Write-Log "Erreur lors du déplacement: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de déplacement" "Impossible de déplacer l'utilisateur :`n$($_.Exception.Message)"
    }
}
