# Système de logging avancé
function Write-Log {
    param(
        [string]$Message,
        [ValidateSet('Info', 'Warning', 'Error', 'Debug')]
        [string]$Level = 'Info'
    )
    
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $logEntry = "[$timestamp] [$Level] $Message"
    
    try {
        Add-Content -Path $script:Config.LogPath -Value $logEntry -ErrorAction SilentlyContinue
    } catch {
        # Fallback si impossible d'écrire dans le log
    }
    
    # Affichage console selon le niveau
    switch ($Level) {
        'Error' { Write-Host $logEntry -ForegroundColor Red }
        'Warning' { Write-Host $logEntry -ForegroundColor Yellow }
        'Info' { Write-Host $logEntry -ForegroundColor Green }
        'Debug' { Write-Host $logEntry -ForegroundColor Cyan }
    }
}

# Gestion des paramètres avec validation
function Load-FavoriteCity {
    try {
        if (Test-Path $script:Config.SettingsPath) {
            $json = Get-Content $script:Config.SettingsPath -Raw -ErrorAction Stop | ConvertFrom-Json -ErrorAction Stop
            if ($json.PSObject.Properties.Name -contains 'FavoriteCity') {
                Write-Log "Ville favorite chargée: $($json.FavoriteCity)" -Level Debug
                return $json.FavoriteCity
            }
        }
    } catch {
        Write-Log "Erreur lors du chargement de la ville favorite: $($_.Exception.Message)" -Level Warning
    }
    return $null
}

function Save-FavoriteCity {
    param([string]$City)
    
    $cleanCity = $City -replace "⭐ ", ""
    
    if ([string]::IsNullOrWhiteSpace($cleanCity)) {
        Write-Log "Tentative de sauvegarde d'une ville vide" -Level Warning
        return
    }
    
    try {
        $settings = @{ FavoriteCity = $cleanCity }
        $settings | ConvertTo-Json | Set-Content $script:Config.SettingsPath -ErrorAction Stop
        Write-Log "Ville favorite sauvegardée: $cleanCity" -Level Info
        
        $currentSelection = $script:UIControls.cbCity.SelectedItem
        Load-Cities
        
        $favoriteWithStar = "⭐ $cleanCity"
        if ($script:UIControls.cbCity.Items.Contains($favoriteWithStar)) {
            $script:UIControls.cbCity.SelectedItem = $favoriteWithStar
        }
        
    } catch {
        Write-Log "Erreur lors de la sauvegarde de la ville favorite: $($_.Exception.Message)" -Level Error
    }
}

# Fonction de validation AD avec retry
function Test-ADConnection {
    param([int]$MaxRetries = $script:Config.MaxRetries)
    
    for ($i = 1; $i -le $MaxRetries; $i++) {
        try {
            $domain = Get-ADDomain -ErrorAction Stop
            Write-Log "Connexion AD réussie au domaine: $($domain.DNSRoot)" -Level Info
            return $true
        } catch {
            Write-Log "Tentative $i/$MaxRetries - Erreur de connexion AD: $($_.Exception.Message)" -Level Warning
            if ($i -lt $MaxRetries) {
                Start-Sleep -Seconds 2
            }
        }
    }
    return $false
}

# Génération sécurisée de mot de passe
function New-SecurePassword {
    param([string]$EmployeeID)
    
    if ([string]::IsNullOrWhiteSpace($EmployeeID)) {
        throw "ID employé requis pour générer le mot de passe"
    }
    
    $randomNumbers = -join ((0..9) | Get-Random -Count 6)
    $specialChars = @('!', '@', '#', '$', '%', '&', '*')
    $specialChar = $specialChars | Get-Random
    
    return "Foundever@$EmployeeID$specialChar$randomNumbers"
}

# Fonction pour gérer les valeurs nulles
function Get-ValueOrDefault {
    param(
        [Parameter(ValueFromPipeline)]
        $Value,
        [string]$Default = ""
    )
    
    if ($null -eq $Value -or [string]::IsNullOrWhiteSpace($Value)) {
        return $Default
    }
    return $Value.ToString()
}

# Extraction de la ville depuis le Distinguished Name
function Get-UserCityFromDN {
    param([string]$DistinguishedName)
    
    try {
        # Pattern pour extraire la ville : OU=VilleName,OU=FR,DC=...
        if ($DistinguishedName -match "OU=([^,]+),OU=FR,DC=") {
            $extractedCity = $matches[1]
            
            # Vérifier que cette ville existe dans notre liste
            $root = New-Object System.DirectoryServices.DirectoryEntry("LDAP://$($script:Config.BaseOU)")
            $searcher = New-Object System.DirectoryServices.DirectorySearcher($root)
            $searcher.Filter = "(&(objectCategory=organizationalUnit)(ou=$extractedCity))"
            $searcher.SearchScope = [System.DirectoryServices.SearchScope]::OneLevel
            $result = $searcher.FindOne()
            
            if ($result) {
                Write-Log "Ville détectée depuis DN : $extractedCity" -Level Debug
                return $extractedCity
            } else {
                Write-Log "Ville '$extractedCity' non trouvée dans la liste des villes disponibles" -Level Warning
            }
        } else {
            Write-Log "Impossible d'extraire la ville depuis le DN : $DistinguishedName" -Level Warning
        }
        
        return $null
        
    } catch {
        Write-Log "Erreur lors de l'extraction de la ville : $($_.Exception.Message)" -Level Warning
        return $null
    }
}

# Synchronisation de la ville dans le ComboBox principal
function Sync-MainCitySelection {
    param([string]$CityName)
    
    try {
        if ([string]::IsNullOrWhiteSpace($CityName)) {
            return
        }
        
        # Chercher la ville avec ou sans étoile dans le ComboBox principal
        $cityWithStar = "⭐ $CityName"
        $cityFound = $false
        
        # Vérifier d'abord avec l'étoile (ville favorite)
        if ($script:UIControls.cbCity.Items.Contains($cityWithStar)) {
            $script:UIControls.cbCity.SelectedItem = $cityWithStar
            $cityFound = $true
            Write-Log "Ville favorite sélectionnée automatiquement : $CityName" -Level Info
        }
        # Sinon chercher sans étoile
        elseif ($script:UIControls.cbCity.Items.Contains($CityName)) {
            $script:UIControls.cbCity.SelectedItem = $CityName
            $cityFound = $true
            Write-Log "Ville sélectionnée automatiquement : $CityName" -Level Info
        }
        
        if ($cityFound) {
            # Déclencher manuellement l'événement de changement de ville
            try {
                $script:UIControls.lbUsers.Items.Clear()
                Clear-UserDetails
                Load-ComputersWithoutSerial
                Write-Log "Interface mise à jour automatiquement pour la ville : $CityName" -Level Info
            } catch {
                Write-Log "Erreur lors de la mise à jour de l'interface : $($_.Exception.Message)" -Level Warning
            }
        } else {
            Write-Log "Ville '$CityName' non trouvée dans le ComboBox principal" -Level Warning
        }
        
    } catch {
        Write-Log "Erreur lors de la synchronisation de la ville : $($_.Exception.Message)" -Level Warning
    }
}

# Clear-UserDetails
function Clear-UserDetails {
    try {
        # Effacer tous les champs de détails
        foreach ($key in $script:UIControls.detail.Keys) {
            if ($script:UIControls.detail[$key]) {
                if ($script:UIControls.detail[$key] -is [System.Windows.Forms.TextBox]) {
                    $script:UIControls.detail[$key].Text = ""
                } elseif ($script:UIControls.detail[$key] -is [System.Windows.Forms.Label]) {
                    $script:UIControls.detail[$key].Text = ""
                    $script:UIControls.detail[$key].ForeColor = $script:Theme.Text
                }
            }
        }
        
        # Effacer la liste des groupes
        if ($script:UIControls.lbGrps) {
            $script:UIControls.lbGrps.Items.Clear()
        }
        
        # Réinitialiser les boutons
        if ($script:UIControls.btnDisable) {
            $script:UIControls.btnDisable.Text = "Désactiver compte"
            $script:UIControls.btnDisable.BackColor = $script:Theme.ButtonSecondary
        }
        
        if ($script:UIControls.btnReset) {
            $script:UIControls.btnReset.Enabled = $true
        }
        
        if ($script:UIControls.btnMoveUser) {
            $script:UIControls.btnMoveUser.Enabled = $true
        }
        
        Write-Log "Détails utilisateur effacés" -Level Debug
        
    } catch {
        Write-Log "Erreur lors de l'effacement des détails: $($_.Exception.Message)" -Level Warning
    }
}

# Réinitialisation avec presse-papier et choix "changement à l'ouverture de session"
function Reset-UserPassword {
    try {
        if (-not $script:UIControls.lbUsers.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un utilisateur."
            return
        }
        
        $selectedText = $script:UIControls.lbUsers.SelectedItem.ToString()
        if ($selectedText -match '\[([^\]]+)\]') {
            $samAccountName = $matches[1]
        } else {
            Show-ErrorMessage "Erreur de sélection" "Impossible d'identifier l'utilisateur sélectionné."
            return
        }
        
        $user = Get-ADUser -Identity $samAccountName -Properties EmployeeID, Enabled, Surname, GivenName -ErrorAction Stop
        
        if ([string]::IsNullOrWhiteSpace($user.EmployeeID)) {
            Show-ErrorMessage "Matricule manquant" "Le matricule est requis pour réinitialiser le mot de passe de $samAccountName."
            return
        }
        
        # Dialogue pour choisir le changement à l'ouverture de session
        $changeAtLogonForm = New-Object System.Windows.Forms.Form -Property @{
            Text = "Options de réinitialisation"
            Size = [System.Drawing.Size]::new(450, 250)
            StartPosition = "CenterParent"
            FormBorderStyle = 'FixedDialog'
            MaximizeBox = $false
            MinimizeBox = $false
            BackColor = $script:Theme.Background
        }
        
        $lblQuestion = New-Object System.Windows.Forms.Label -Property @{
            Text = "Forcer le changement de mot de passe à l'ouverture de session ?"
            Location = [System.Drawing.Point]::new(20, 20)
            Size = [System.Drawing.Size]::new(400, 40)
            Font = [System.Drawing.Font]::new("Segoe UI", 10, [System.Drawing.FontStyle]::Bold)
            ForeColor = $script:Theme.Primary
            TextAlign = 'MiddleCenter'
        }
        
        $lblUser = New-Object System.Windows.Forms.Label -Property @{
            Text = "Utilisateur : $($user.GivenName) $($user.Surname) [$samAccountName]"
            Location = [System.Drawing.Point]::new(20, 70)
            Size = [System.Drawing.Size]::new(400, 20)
            Font = [System.Drawing.Font]::new("Segoe UI", 9)
            ForeColor = $script:Theme.Text
            TextAlign = 'MiddleCenter'
        }
        
        $chkChangeAtLogon = New-Object System.Windows.Forms.CheckBox -Property @{
            Text = "L'utilisateur doit changer le mot de passe à l'ouverture de session"
            Location = [System.Drawing.Point]::new(20, 110)
            Size = [System.Drawing.Size]::new(400, 30)
            Font = [System.Drawing.Font]::new("Segoe UI", 9)
            ForeColor = $script:Theme.Text
            Checked = $true
        }
        
        $btnOK = New-Object System.Windows.Forms.Button -Property @{
            Text = "Réinitialiser"
            Location = [System.Drawing.Point]::new(250, 160)
            Size = [System.Drawing.Size]::new(100, 30)
            BackColor = $script:Theme.ButtonPrimary
            ForeColor = $script:Theme.White
            FlatStyle = 'Flat'
            DialogResult = [System.Windows.Forms.DialogResult]::OK
        }
        
        $btnCancel = New-Object System.Windows.Forms.Button -Property @{
            Text = "Annuler"
            Location = [System.Drawing.Point]::new(360, 160)
            Size = [System.Drawing.Size]::new(70, 30)
            BackColor = $script:Theme.ButtonSecondary
            ForeColor = $script:Theme.White
            FlatStyle = 'Flat'
            DialogResult = [System.Windows.Forms.DialogResult]::Cancel
        }
        
        $changeAtLogonForm.Controls.AddRange(@($lblQuestion, $lblUser, $chkChangeAtLogon, $btnOK, $btnCancel))
        
        $result = $changeAtLogonForm.ShowDialog()
        
        if ($result -ne [System.Windows.Forms.DialogResult]::OK) {
            return
        }
        
        $forceChangeAtLogon = $chkChangeAtLogon.Checked
        
        if (-not $user.Enabled) {
            Set-ADUser -Identity $samAccountName -Enabled $true
            Write-Log "Compte $samAccountName activé" -Level Info
        }
        
        $newPassword = New-SecurePassword $user.EmployeeID
        $securePassword = ConvertTo-SecureString $newPassword -AsPlainText -Force
        
        Set-ADAccountPassword -Identity $samAccountName -Reset -NewPassword $securePassword
        
        # Appliquer ou non le changement à l'ouverture de session
        if ($forceChangeAtLogon) {
            Set-ADUser -Identity $samAccountName -ChangePasswordAtLogon $true
            Write-Log "Changement de mot de passe à l'ouverture de session activé pour $samAccountName" -Level Info
        } else {
            Set-ADUser -Identity $samAccountName -ChangePasswordAtLogon $false
            Write-Log "Changement de mot de passe à l'ouverture de session désactivé pour $samAccountName" -Level Info
        }
        
        # Copie automatique dans le presse-papier
        try {
            [System.Windows.Forms.Clipboard]::SetText($newPassword)
            $clipboardStatus = "✓ Mot de passe copié dans le presse-papier"
            Write-Log "Mot de passe copié dans le presse-papier pour $samAccountName" -Level Info
        } catch {
            $clipboardStatus = "⚠ Impossible de copier dans le presse-papier"
            Write-Log "Erreur lors de la copie dans le presse-papier: $($_.Exception.Message)" -Level Warning
        }
        
        # Message personnalisé avec statut du presse-papier
        $changeMessage = if ($forceChangeAtLogon) {
            "Il sera à modifier à l'ouverture de session."
        } else {
            "Le mot de passe est définitif."
        }
        
        $successMessage = "Bonjour,`n" +
                         "Le mot de passe de $($user.Surname) $($user.GivenName) est réinitialisé : $newPassword`n" +
                         "$changeMessage`n`n" +
                         "$clipboardStatus"
        
        Show-InfoMessage $successMessage "Réinitialisation réussie"
        
        Write-Log "Mot de passe réinitialisé pour $samAccountName (changement à l'ouverture: $forceChangeAtLogon)" -Level Info
        Show-UserDetails
        
    } catch {
        Write-Log "Erreur lors de la réinitialisation du mot de passe: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de réinitialisation" "Impossible de réinitialiser le mot de passe: $($_.Exception.Message)"
    }
}

# Chargement des ordinateurs filtrés par ville sélectionnée
function Load-ComputersWithoutSerial {
    try {
        $script:UIControls.lbComputers.Items.Clear()
        
        $selectedCity = ""
        if ($script:UIControls.cbCity.SelectedItem) {
            $selectedCityRaw = $script:UIControls.cbCity.SelectedItem.ToString()
            $selectedCity = $selectedCityRaw -replace "⭐ ", ""
        }
        
        if ([string]::IsNullOrWhiteSpace($selectedCity)) {
            $script:UIControls.lbComputers.Items.Add("Veuillez sélectionner une ville")
            $script:UIControls.grpComputers.Text = "PC sans N° série - Aucune ville"
            return
        }
        
        $searchBase = "OU=$selectedCity,$($script:Config.BaseOU)"
        
        try {
            $null = Get-ADOrganizationalUnit -Identity $searchBase -ErrorAction Stop
        } catch {
            $script:UIControls.lbComputers.Items.Add("Ville introuvable : $selectedCity")
            $script:UIControls.grpComputers.Text = "PC sans N° série - Erreur"
            return
        }
        
        # Recherche améliorée avec détection des numéros de série invalides
        $computers = Get-ADComputer -SearchBase $searchBase -Filter "Enabled -eq 'True'" -Properties Name, SerialNumber, Description, OperatingSystem, LastLogonDate |
                    Where-Object { 
                        [string]::IsNullOrWhiteSpace($_.SerialNumber) -or 
                        $_.SerialNumber -eq "To be filled by O.E.M." -or 
                        $_.SerialNumber -eq "Default string" -or 
                        $_.SerialNumber -eq "System Serial Number" -or
                        $_.SerialNumber -eq "Not Specified" -or
                        $_.SerialNumber -eq "0" -or
                        $_.SerialNumber.Length -lt 3
                    } |
                    Sort-Object Name |
                    Select-Object -First 100
        
        if ($computers.Count -eq 0) {
            $script:UIControls.lbComputers.Items.Add("Tous les PC ont un N° série valide")
        } else {
            foreach ($computer in $computers) {
                $displayName = $computer.Name
                
                if (-not [string]::IsNullOrWhiteSpace($computer.OperatingSystem)) {
                    $osShort = $computer.OperatingSystem -replace "Microsoft Windows ", "Win"
                    $displayName += " [$osShort]"
                }
                
                if ($computer.LastLogonDate -and $computer.LastLogonDate -gt (Get-Date).AddDays(-7)) {
                    $displayName += " [RECENT]"
                } else {
                    $displayName += " [ACTIF]"
                }
                
                $script:UIControls.lbComputers.Items.Add($displayName)
            }
        }
        
        $timestamp = Get-Date -Format "HH:mm:ss"
        $script:UIControls.grpComputers.Text = "PC sans N° série - $selectedCity ($($computers.Count)) - $timestamp"
        
    } catch {
        Write-Log "Erreur lors du chargement des ordinateurs: $($_.Exception.Message)" -Level Error
        $script:UIControls.lbComputers.Items.Clear()
        $script:UIControls.lbComputers.Items.Add("Erreur de chargement")
    }
}

# Mise à jour du numéro de série dans AD avec rafraîchissement immédiat
function Update-ComputerSerialInAD {
    param(
        [string]$ComputerName,
        [string]$SerialNumber
    )
    
    try {
        # Vérifier que l'ordinateur existe
        $computer = Get-ADComputer -Identity $ComputerName -ErrorAction Stop
        
        $confirmMessage = "Mettre à jour le numéro de série ?`n`n" +
                         "Ordinateur : $ComputerName`n" +
                         "Nouveau numéro de série : $SerialNumber`n`n" +
                         "Cette action remplacera le numéro de série existant."
        
        if (-not (Show-ConfirmDialog $confirmMessage "Confirmation de mise à jour")) {
            return
        }
        
        # REMPLACER le numéro de série existant
        Set-ADComputer -Identity $ComputerName -Replace @{SerialNumber = $SerialNumber} -ErrorAction Stop
        
        Write-Log "Numéro de série mis à jour pour $ComputerName : $SerialNumber" -Level Info
        
        Show-InfoMessage "Numéro de série mis à jour avec succès !`n`nOrdinateur : $ComputerName`nNuméro de série : $SerialNumber" "Mise à jour réussie"
        
        # Rafraîchissement immédiat de la liste
        Load-ComputersWithoutSerial
        
    } catch {
        Write-Log "Erreur lors de la mise à jour du numéro de série: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de mise à jour" "Impossible de mettre à jour le numéro de série dans Active Directory:`n$($_.Exception.Message)"
    }
}
