# FONCTION CORRIGÉE : Recherche d'utilisateurs pour gestion en lot avec tri par ville
function Search-BulkUsers {
    param(
        [string]$SearchTerm,
        [System.Windows.Forms.CheckedListBox]$UserList,
        [System.Windows.Forms.Label]$CountLabel
    )
    
    try {
        $UserList.Items.Clear()
        
        if ([string]::IsNullOrWhiteSpace($SearchTerm) -or $SearchTerm.Length -lt 2) {
            Show-InfoMessage "Veuillez saisir au moins 2 caractères pour la recherche."
            return
        }
        
        # Récupération de la ville sélectionnée dans l'interface principale
        $selectedCity = ""
        if ($script:UIControls.cbCity.SelectedItem) {
            $selectedCityRaw = $script:UIControls.cbCity.SelectedItem.ToString()
            $selectedCity = $selectedCityRaw -replace "⭐ ", ""
        }
        
        $allUsers = @()
        
        if (-not [string]::IsNullOrWhiteSpace($selectedCity)) {
            # Recherche UNIQUEMENT dans la ville sélectionnée
            try {
                $searchBase = "OU=$selectedCity,$($script:Config.BaseOU)"
                $users = Get-ADUser -SearchBase $searchBase -Filter "GivenName -like '*$SearchTerm*' -or Surname -like '*$SearchTerm*' -or SamAccountName -like '*$SearchTerm*'" -Properties GivenName, Surname, SamAccountName, Enabled, Department, DistinguishedName -ErrorAction Stop |
                         Where-Object { $_.Enabled -eq $true } |
                         Sort-Object Surname, GivenName |
                         Select-Object -First 100
                
                foreach ($user in $users) {
                    $displayName = "$($user.Surname) $($user.GivenName) [$($user.SamAccountName)] - $selectedCity"
                    if ($user.Department) {
                        $displayName += " ($($user.Department))"
                    }
                    $allUsers += $displayName
                }
                
                Write-Log "Recherche en lot dans $selectedCity : $($users.Count) utilisateurs trouvés pour '$SearchTerm'" -Level Info
            } catch {
                Write-Log "Erreur recherche dans $selectedCity : $($_.Exception.Message)" -Level Warning
                Show-ErrorMessage "Erreur de recherche" "Impossible de rechercher dans la ville $selectedCity : $($_.Exception.Message)"
                return
            }
        } else {
            # Si aucune ville sélectionnée, recherche dans toutes les villes
            $cities = @()
            
            try {
                $root = New-Object System.DirectoryServices.DirectoryEntry("LDAP://$($script:Config.BaseOU)")
                $searcher = New-Object System.DirectoryServices.DirectorySearcher($root)
                $searcher.Filter = "(objectCategory=organizationalUnit)"
                $searcher.SearchScope = [System.DirectoryServices.SearchScope]::OneLevel
                $searcher.PropertiesToLoad.Add("ou") | Out-Null
                
                $cities = $searcher.FindAll() | ForEach-Object {
                    if ($_.Properties["ou"].Count -gt 0) { 
                        $_.Properties["ou"][0] 
                    }
                } | Sort-Object
                
                $searcher.Dispose()
                $root.Dispose()
            } catch {
                Write-Log "Erreur lors de la récupération des villes : $($_.Exception.Message)" -Level Error
                Show-ErrorMessage "Erreur" "Impossible de récupérer la liste des villes : $($_.Exception.Message)"
                return
            }
            
            # Recherche dans chaque ville avec limitation
            foreach ($city in $cities) {
                try {
                    $searchBase = "OU=$city,$($script:Config.BaseOU)"
                    $users = Get-ADUser -SearchBase $searchBase -Filter "GivenName -like '*$SearchTerm*' -or Surname -like '*$SearchTerm*' -or SamAccountName -like '*$SearchTerm*'" -Properties GivenName, Surname, SamAccountName, Enabled, Department -ErrorAction Stop |
                             Where-Object { $_.Enabled -eq $true } |
                             Sort-Object Surname, GivenName |
                             Select-Object -First 10  # Limitation par ville
                    
                    foreach ($user in $users) {
                        $displayName = "$($user.Surname) $($user.GivenName) [$($user.SamAccountName)] - $city"
                        if ($user.Department) {
                            $displayName += " ($($user.Department))"
                        }
                        $allUsers += $displayName
                    }
                } catch {
                    Write-Log "Erreur recherche dans $city : $($_.Exception.Message)" -Level Warning
                }
            }
        }
        
        # Tri final par ville puis par nom
        $allUsers = $allUsers | Sort-Object { 
            $parts = $_ -split ' - '
            if ($parts.Count -ge 2) {
                "$($parts[1])_$($parts[0])"  # Ville_Nom pour tri
            } else {
                $_
            }
        }
        
        # Ajout à la liste
        foreach ($user in $allUsers) {
            $UserList.Items.Add($user)
        }
        
        $cityInfo = if (-not [string]::IsNullOrWhiteSpace($selectedCity)) { " dans $selectedCity" } else { " (toutes villes)" }
        $CountLabel.Text = "$($allUsers.Count) utilisateur(s) trouvé(s)$cityInfo"
        $CountLabel.ForeColor = if ($allUsers.Count -gt 0) { [System.Drawing.Color]::Green } else { [System.Drawing.Color]::Red }
        
    } catch {
        Write-Log "Erreur lors de la recherche d'utilisateurs en lot: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur de recherche" "Impossible de rechercher les utilisateurs: $($_.Exception.Message)"
        $CountLabel.Text = "Erreur lors de la recherche"
        $CountLabel.ForeColor = [System.Drawing.Color]::Red
    }
}

# FONCTION CORRIGÉE : Mise à jour du compteur d'utilisateurs sélectionnés
function Update-UserSelectionCount {
    param(
        [System.Windows.Forms.CheckedListBox]$UserList,
        [System.Windows.Forms.Label]$CountLabel
    )
    
    try {
        if ($UserList -and $CountLabel) {
            $checkedCount = $UserList.CheckedItems.Count
            $CountLabel.Text = "$checkedCount utilisateur(s) sélectionné(s) sur $($UserList.Items.Count)"
            
            if ($checkedCount -gt 0) {
                $CountLabel.ForeColor = [System.Drawing.Color]::Green
            } else {
                $CountLabel.ForeColor = $script:Theme.Text
            }
        }
    } catch {
        Write-Log "Erreur lors de la mise à jour du compteur: $($_.Exception.Message)" -Level Warning
    }
}









# FONCTION RENFORCÉE : Exécution des opérations en lot avec gestion d'erreurs complète
function Execute-BulkGroupOperation {
    param(
        [System.Windows.Forms.CheckedListBox]$UserList,
        [System.Windows.Forms.ListBox]$GroupList,
        [System.Windows.Forms.TextBox]$ResultsBox,
        [string]$Operation
    )
    
    try {
        if (-not $UserList -or -not $GroupList -or -not $ResultsBox) {
            Show-ErrorMessage "Erreur de paramètres" "Paramètres manquants pour l'opération en lot."
            return
        }
        
        if ($UserList.CheckedItems.Count -eq 0) {
            Show-InfoMessage "Veuillez sélectionner au moins un utilisateur."
            return
        }
        
        if (-not $GroupList.SelectedItem) {
            Show-InfoMessage "Veuillez sélectionner un groupe."
            return
        }
        
        # Extraction sécurisée du nom du groupe
        $selectedGroup = $GroupList.SelectedItem.ToString()
        $groupName = ""
        
        if ($selectedGroup -match '^([^-]+)') {
            $groupName = $matches[1].Trim()
        } else {
            $groupName = $selectedGroup.Trim()
        }
        
        if ([string]::IsNullOrWhiteSpace($groupName)) {
            Show-ErrorMessage "Erreur de groupe" "Impossible d'identifier le nom du groupe sélectionné."
            return
        }
        
        # Vérification de l'existence du groupe
        try {
            $null = Get-ADGroup -Identity $groupName -ErrorAction Stop
        } catch {
            Show-ErrorMessage "Groupe introuvable" "Le groupe '$groupName' n'existe pas dans Active Directory."
            return
        }
        
        # Extraction sécurisée des utilisateurs sélectionnés
        $selectedUsers = @()
        foreach ($checkedItem in $UserList.CheckedItems) {
            try {
                $itemText = $checkedItem.ToString()
                if ($itemText -match '\[([^\]]+)\]') {
                    $samAccountName = $matches[1].Trim()
                    if (-not [string]::IsNullOrWhiteSpace($samAccountName)) {
                        $selectedUsers += $samAccountName
                    }
                }
            } catch {
                Write-Log "Erreur lors de l'extraction de l'utilisateur: $itemText" -Level Warning
            }
        }
        
        if ($selectedUsers.Count -eq 0) {
            Show-ErrorMessage "Aucun utilisateur valide" "Aucun utilisateur valide n'a pu être extrait de la sélection."
            return
        }
        
        $operationText = if ($Operation -eq "add") { "AJOUT" } else { "SUPPRESSION" }
        $actionText = if ($Operation -eq "add") { "ajouter" } else { "retirer" }
        $preposition = if ($Operation -eq "add") { "au" } else { "du" }
        
        $confirmMessage = "$operationText EN LOT`n`n" +
                         "Vous êtes sur le point de $actionText $($selectedUsers.Count) utilisateur(s) " +
                         "$preposition groupe '$groupName'.`n`n" +
                         "Utilisateurs concernés :`n" +
                         ($selectedUsers | ForEach-Object { "• $_" } | Select-Object -First 10 | Out-String) +
                         $(if ($selectedUsers.Count -gt 10) { "`n... et $($selectedUsers.Count - 10) autre(s)`n" } else { "" }) +
                         "`nContinuer ?"
        
        if (-not (Show-ConfirmDialog $confirmMessage "Confirmation $operationText")) {
            return
        }
        
        # Exécution des opérations avec gestion d'erreurs robuste
        $results = @()
        $successCount = 0
        $errorCount = 0
        $alreadyCount = 0
        
        $results += "=== $operationText EN LOT - $(Get-Date -Format 'dd/MM/yyyy HH:mm:ss') ==="
        $results += "Groupe : $groupName"
        $results += "Utilisateurs à traiter : $($selectedUsers.Count)"
        $results += ""
        
        foreach ($samAccountName in $selectedUsers) {
            try {
                # Vérification de l'existence de l'utilisateur
                $user = $null
                try {
                    $user = Get-ADUser -Identity $samAccountName -ErrorAction Stop
                } catch {
                    $results += "ERREUR - $samAccountName : Utilisateur introuvable"
                    $errorCount++
                    continue
                }
                
                if ($Operation -eq "add") {
                    # Vérifier si déjà membre
                    try {
                        $currentGroups = Get-ADPrincipalGroupMembership -Identity $samAccountName -ErrorAction Stop | Select-Object -ExpandProperty Name
                        if ($currentGroups -contains $groupName) {
                            $results += "DÉJÀ MEMBRE - $samAccountName : Déjà dans le groupe"
                            $alreadyCount++
                        } else {
                            Add-ADGroupMember -Identity $groupName -Members $samAccountName -ErrorAction Stop
                            $results += "SUCCÈS - $samAccountName : Ajouté au groupe"
                            $successCount++
                        }
                    } catch {
                        $results += "ERREUR - $samAccountName : $($_.Exception.Message)"
                        $errorCount++
                    }
                } else {
                    # Opération de suppression
                    try {
                        $currentGroups = Get-ADPrincipalGroupMembership -Identity $samAccountName -ErrorAction Stop | Select-Object -ExpandProperty Name
                        if ($currentGroups -notcontains $groupName) {
                            $results += "PAS MEMBRE - $samAccountName : N'était pas dans le groupe"
                            $alreadyCount++
                        } else {
                            Remove-ADGroupMember -Identity $groupName -Members $samAccountName -Confirm:$false -ErrorAction Stop
                            $results += "SUCCÈS - $samAccountName : Retiré du groupe"
                            $successCount++
                        }
                    } catch {
                        $results += "ERREUR - $samAccountName : $($_.Exception.Message)"
                        $errorCount++
                    }
                }
            } catch {
                $results += "ERREUR CRITIQUE - $samAccountName : $($_.Exception.Message)"
                $errorCount++
            }
        }
        
        $results += ""
        $results += "=== RÉSUMÉ ==="
        $results += "Succès : $successCount"
        $results += "Déjà traités : $alreadyCount"
        $results += "Erreurs : $errorCount"
        $results += "Total traité : $($selectedUsers.Count)"
        $results += "================================="
        $results += ""
        
        # Affichage des résultats
        try {
            $ResultsBox.Text = ($results -join "`r`n")
        } catch {
            Write-Log "Erreur lors de l'affichage des résultats" -Level Warning
        }
        
        # Message de fin
        $summaryMessage = "$operationText TERMINÉ`n`n" +
                         "Groupe : $groupName`n" +
                         "Utilisateurs traités : $($selectedUsers.Count)`n" +
                         "Succès : $successCount`n" +
                         "Déjà traités : $alreadyCount`n" +
                         "Erreurs : $errorCount"
        
        Show-InfoMessage $summaryMessage "Opération terminée"
        
        Write-Log "$operationText en lot terminé - Groupe: $groupName, Succès: $successCount, Déjà traités: $alreadyCount, Erreurs: $errorCount" -Level Info
        
    } catch {
        Write-Log "Erreur lors de l'opération en lot: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur d'opération" "Impossible d'effectuer l'opération en lot: $($_.Exception.Message)"
    }
}
