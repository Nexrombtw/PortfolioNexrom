# Système de logging avancé
function Write-Log {
    param(
        [string]$Message,
        [ValidateSet('Info', 'Warning', 'Error', 'Debug')]
        [string]$Level = 'Info'
    )
    
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $logEntry = "[$timestamp] [$Level] $Message"
    
    try {
        Add-Content -Path $script:Config.LogPath -Value $logEntry -ErrorAction SilentlyContinue
    } catch {
        # Fallback si impossible d'écrire dans le log
    }
    
    # Affichage console selon le niveau
    switch ($Level) {
        'Error' { Write-Host $logEntry -ForegroundColor Red }
        'Warning' { Write-Host $logEntry -ForegroundColor Yellow }
        'Info' { Write-Host $logEntry -ForegroundColor Green }
        'Debug' { Write-Host $logEntry -ForegroundColor Cyan }
    }
}

# Gestion des paramètres avec validation
function Load-FavoriteCity {
    try {
        if (Test-Path $script:Config.SettingsPath) {
            $json = Get-Content $script:Config.SettingsPath -Raw -ErrorAction Stop | ConvertFrom-Json -ErrorAction Stop
            if ($json.PSObject.Properties.Name -contains 'FavoriteCity') {
                Write-Log "Ville favorite chargée: $($json.FavoriteCity)" -Level Debug
                return $json.FavoriteCity
            }
        }
    } catch {
        Write-Log "Erreur lors du chargement de la ville favorite: $($_.Exception.Message)" -Level Warning
    }
    return $null
}

function Save-FavoriteCity {
    param([string]$City)
    
    $cleanCity = $City -replace "⭐ ", ""
    
    if ([string]::IsNullOrWhiteSpace($cleanCity)) {
        Write-Log "Tentative de sauvegarde d'une ville vide" -Level Warning
        return
    }
    
    try {
        $settings = @{ FavoriteCity = $cleanCity }
        $settings | ConvertTo-Json | Set-Content $script:Config.SettingsPath -ErrorAction Stop
        Write-Log "Ville favorite sauvegardée: $cleanCity" -Level Info
        
        $currentSelection = $script:UIControls.cbCity.SelectedItem
        Load-Cities
        
        $favoriteWithStar = "⭐ $cleanCity"
        if ($script:UIControls.cbCity.Items.Contains($favoriteWithStar)) {
            $script:UIControls.cbCity.SelectedItem = $favoriteWithStar
        }
        
    } catch {
        Write-Log "Erreur lors de la sauvegarde de la ville favorite: $($_.Exception.Message)" -Level Error
    }
}

# Fonction de validation AD avec retry
function Test-ADConnection {
    param([int]$MaxRetries = $script:Config.MaxRetries)
    
    for ($i = 1; $i -le $MaxRetries; $i++) {
        try {
            $domain = Get-ADDomain -ErrorAction Stop
            Write-Log "Connexion AD réussie au domaine: $($domain.DNSRoot)" -Level Info
            return $true
        } catch {
            Write-Log "Tentative $i/$MaxRetries - Erreur de connexion AD: $($_.Exception.Message)" -Level Warning
            if ($i -lt $MaxRetries) {
                Start-Sleep -Seconds 2
            }
        }
    }
    return $false
}

# Validation et nettoyage des entrées utilisateur
function Get-SafeInput {
    param(
        [string]$Input,
        [ValidateSet('Username', 'Search', 'General')]
        [string]$Type = 'General'
    )
    
    if ([string]::IsNullOrWhiteSpace($Input)) {
        return ""
    }
    
    $cleaned = $Input.Trim()
    
    switch ($Type) {
        'Username' {
            $cleaned = $cleaned -replace '[<>:"/\\|?*]', ''
            $cleaned = $cleaned -replace '\s+', ' '
        }
        'Search' {
            $cleaned = $cleaned -replace '[\(\)\*\\]', ''
        }
    }
    
    return $cleaned.Substring(0, [Math]::Min($cleaned.Length, 256))
}

# Génération sécurisée de mot de passe
function New-SecurePassword {
    param([string]$EmployeeID)
    
    if ([string]::IsNullOrWhiteSpace($EmployeeID)) {
        throw "ID employé requis pour générer le mot de passe"
    }
    
    $randomNumbers = -join ((0..9) | Get-Random -Count 6)
    $specialChars = @('!', '@', '#', '$', '%', '&', '*')
    $specialChar = $specialChars | Get-Random
    
    return "Foundever@$EmployeeID$specialChar$randomNumbers"
}

# Fonction pour gérer les valeurs nulles
function Get-ValueOrDefault {
    param(
        [Parameter(ValueFromPipeline)]
        $Value,
        [string]$Default = ""
    )
    
    if ($null -eq $Value -or [string]::IsNullOrWhiteSpace($Value)) {
        return $Default
    }
    return $Value.ToString()
}

# Fonction pour gérer les raccourcis clavier globaux
function Initialize-KeyboardShortcuts {
    try {
        # Activer la capture des touches au niveau du formulaire
        $script:UIControls.Form.KeyPreview = $true
        
        # Gestionnaire d'événements pour les raccourcis clavier
        $script:UIControls.Form.Add_KeyDown({
            param($sender, $e)
            
            # Ctrl+F : Focus sur le champ de recherche
            if ($e.Control -and $e.KeyCode -eq 'F') {
                if ($script:UIControls.MainTabControl.SelectedIndex -eq 0) {
                    $script:UIControls.txtSearch.Focus()
                    $script:UIControls.txtSearch.SelectAll()
                }
                $e.SuppressKeyPress = $true
                Write-Log "Raccourci Ctrl+F activé - Focus sur recherche" -Level Debug
            }
            
            # F5 : Actualiser selon l'onglet actif
            elseif ($e.KeyCode -eq 'F5') {
                try {
                    switch ($script:UIControls.MainTabControl.SelectedIndex) {
                        0 { # Onglet Utilisateurs
                            Refresh-Users
                            Write-Log "F5 - Liste utilisateurs actualisée" -Level Info
                        }
                        1 { # Onglet Groupes
                            Refresh-GroupsList
                            Write-Log "F5 - Liste groupes actualisée" -Level Info
                        }
                        2 { # Onglet Rapports
                            Refresh-Reports
                            Write-Log "F5 - Rapports actualisés" -Level Info
                        }
                        default {
                            Write-Log "F5 - Onglet non supporté pour l'actualisation" -Level Warning
                        }
                    }
                } catch {
                    Show-ErrorMessage "Erreur d'actualisation" $_.Exception.Message
                }
                $e.SuppressKeyPress = $true
            }
            
            # Ctrl+R : Réinitialiser mot de passe (onglet utilisateurs uniquement)
            elseif ($e.Control -and $e.KeyCode -eq 'R') {
                if ($script:UIControls.MainTabControl.SelectedIndex -eq 0 -and $script:UIControls.lbUsers.SelectedItem) {
                    try {
                        Reset-UserPassword
                        Write-Log "Raccourci Ctrl+R activé - Réinitialisation mot de passe" -Level Info
                    } catch {
                        Show-ErrorMessage "Erreur de réinitialisation" $_.Exception.Message
                    }
                }
                $e.SuppressKeyPress = $true
            }
            
            # Ctrl+D : Activer/Désactiver compte (onglet utilisateurs uniquement)
            elseif ($e.Control -and $e.KeyCode -eq 'D') {
                if ($script:UIControls.MainTabControl.SelectedIndex -eq 0 -and $script:UIControls.lbUsers.SelectedItem) {
                    try {
                        if ($script:UIControls.btnDisable.Text -eq "Désactiver compte") {
                            Disable-UserAccount
                        } else {
                            Enable-UserAccount
                        }
                        Write-Log "Raccourci Ctrl+D activé - Basculement état compte" -Level Info
                    } catch {
                        Show-ErrorMessage "Erreur d'opération" $_.Exception.Message
                    }
                }
                $e.SuppressKeyPress = $true
            }
            
            # Ctrl+G : Focus sur recherche de groupes
            elseif ($e.Control -and $e.KeyCode -eq 'G') {
                if ($script:UIControls.MainTabControl.SelectedIndex -eq 0) {
                    $script:UIControls.txtGroupSearch.Focus()
                    $script:UIControls.txtGroupSearch.SelectAll()
                } elseif ($script:UIControls.MainTabControl.SelectedIndex -eq 1) {
                    $script:UIControls.txtGroupsSearch.Focus()
                    $script:UIControls.txtGroupsSearch.SelectAll()
                }
                $e.SuppressKeyPress = $true
                Write-Log "Raccourci Ctrl+G activé - Focus sur recherche groupes" -Level Debug
            }
            
            # Ctrl+Tab : Changer d'onglet vers la droite
            elseif ($e.Control -and $e.KeyCode -eq 'Tab') {
                $currentIndex = $script:UIControls.MainTabControl.SelectedIndex
                $nextIndex = ($currentIndex + 1) % $script:UIControls.MainTabControl.TabCount
                $script:UIControls.MainTabControl.SelectedIndex = $nextIndex
                $e.SuppressKeyPress = $true
                Write-Log "Raccourci Ctrl+Tab - Changement d'onglet vers $nextIndex" -Level Debug
            }
            
            # Échap : Effacer les sélections selon l'onglet
            elseif ($e.KeyCode -eq 'Escape') {
                switch ($script:UIControls.MainTabControl.SelectedIndex) {
                    0 { # Onglet Utilisateurs
                        $script:UIControls.txtSearch.Text = ""
                        $script:UIControls.lbUsers.ClearSelected()
                        Clear-UserDetails
                    }
                    1 { # Onglet Groupes
                        $script:UIControls.txtGroupsSearch.Text = ""
                        $script:UIControls.lbAllGroups.ClearSelected()
                    }
                }
                $e.SuppressKeyPress = $true
                Write-Log "Raccourci Échap activé - Sélections effacées" -Level Debug
            }
        })
        
        Write-Log "Raccourcis clavier initialisés" -Level Info
        return $true
    } catch {
        Write-Log "Erreur lors de l'initialisation des raccourcis clavier: $($_.Exception.Message)" -Level Error
        return $false
    }
}

    }

# Fonction pour actualiser la liste des groupes
function Refresh-GroupsList {
    try {
        $script:UIControls.lbAllGroups.Items.Clear()
        $groups = Get-ADGroup -Filter * | Sort-Object Name | Select-Object -First 1000
        
        foreach ($group in $groups) {
            $script:UIControls.lbAllGroups.Items.Add("$($group.Name) - $($group.GroupScope)")
        }
        
        Write-Log "Liste des groupes actualisée - $($groups.Count) groupes chargés" -Level Info
    } catch {
        Write-Log "Erreur lors de l'actualisation des groupes: $($_.Exception.Message)" -Level Error
        throw
    }
}

# Fonction pour rechercher tous les groupes AD
function Search-AllADGroups {
    try {
        $searchTerm = $script:UIControls.txtGroupsSearch.Text.Trim()
        
        if ([string]::IsNullOrWhiteSpace($searchTerm)) {
            Refresh-GroupsList
            return
        }
        
        $script:UIControls.lbAllGroups.Items.Clear()
        $groups = Get-ADGroup -Filter "Name -like '*$searchTerm*'" | Sort-Object Name | Select-Object -First 500
        
        foreach ($group in $groups) {
            $script:UIControls.lbAllGroups.Items.Add("$($group.Name) - $($group.GroupScope)")
        }
        
        Write-Log "Recherche de groupes terminée - $($groups.Count) groupes trouvés pour '$searchTerm'" -Level Info
    } catch {
        Write-Log "Erreur lors de la recherche de groupes: $($_.Exception.Message)" -Level Error
        throw
    }
}

            "$env:USERPROFILE\Pictures\foundever_logo.png",
            "$env:USERPROFILE\Desktop\foundever_logo.png"
        )
        
        $logoLoaded = $false
        foreach ($logoPath in $logoLocations) {
            if (Test-Path $logoPath) {
                try {
                    $logoImage = [System.Drawing.Image]::FromFile((Get-Item $logoPath))
                    $script:UIControls.logoBox.Image = $logoImage
                    $logoLoaded = $true
                    Write-Log "Logo Foundever chargé depuis: $logoPath" -Level Info
                    break
                } catch {
                    Write-Log "Erreur lors du chargement du logo depuis $logoPath : $($_.Exception.Message)" -Level Warning
                }
            }
        }
        
        # Placeholder si pas de logo
        if (-not $logoLoaded) {
            $script:UIControls.logoBox.BackColor = [System.Drawing.Color]::White
            $script:UIControls.logoBox.BorderStyle = 'FixedSingle'
            
            $logoLabel = New-Object System.Windows.Forms.Label -Property @{
                Text = "FOUNDEVER"
                Location = [System.Drawing.Point]::new(0, 0)
                Size = [System.Drawing.Size]::new(450, 85)
                TextAlign = 'MiddleCenter'
                Font = [System.Drawing.Font]::new("Segoe UI", 16, [System.Drawing.FontStyle]::Bold)
                ForeColor = [System.Drawing.Color]::FromArgb(0, 102, 204)
                BackColor = [System.Drawing.Color]::Transparent
            }
            $script:UIControls.logoBox.Controls.Add($logoLabel)
            Write-Log "Logo Foundever non trouvé - placeholder texte créé" -Level Warning
        }
        
        $script:UIControls.Form.Controls.Add($script:UIControls.logoBox)
        Write-Log "Logo Foundever initialisé" -Level Info
        return $true
        
    } catch {
        Write-Log "Erreur lors de l'initialisation du logo: $($_.Exception.Message)" -Level Error
        return $false
    }
}

# Fonction principale d'initialisation modifiée pour les onglets
function Initialize-Application {
    try {
        Write-Log "=== DÉMARRAGE DE L'APPLICATION AVEC ONGLETS ===" -Level Info
        Write-Log "Version PowerShell: $($PSVersionTable.PSVersion)" -Level Info
        Write-Log "Utilisateur: $env:USERNAME" -Level Info
        Write-Log "Ordinateur: $env:COMPUTERNAME" -Level Info
        
        if (-not (Test-ADConnection)) {
            Show-ErrorMessage "Connexion AD impossible" "Impossible de se connecter à Active Directory. Vérifiez votre connexion réseau et vos permissions."
            return $false
        }
        
        if (-not (Initialize-MainForm)) { return $false }
        if (-not (Initialize-TabbedInterface)) { return $false }  # Interface avec onglets
        if (-not (Initialize-KeyboardShortcuts)) { return $false }  # Raccourcis clavier
        if (-not (Initialize-Logo)) { return $false }
        
        if (-not (Load-Cities)) { return $false }
        
        Write-Log "Application avec onglets et raccourcis initialisée avec succès" -Level Info
        return $true
        
    } catch {
        Write-Log "Erreur critique lors de l'initialisation: $($_.Exception.Message)" -Level Error
        Show-ErrorMessage "Erreur d'initialisation" "Impossible d'initialiser l'application: $($_.Exception.Message)"
        return $false
    }
}
