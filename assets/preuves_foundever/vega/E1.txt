# Chargement des villes via DirectoryServices (ADSI)
$baseOU = "OU=FR,DC=Marche,DC=lan"
$cbCity.Items.Clear()
try {
    $root = New-Object System.DirectoryServices.DirectoryEntry("LDAP://$baseOU")
    $searcher = New-Object System.DirectoryServices.DirectorySearcher($root)
    $searcher.Filter = "(objectCategory=organizationalUnit)"
    $searcher.SearchScope = [System.DirectoryServices.SearchScope]::OneLevel
    $searcher.PropertiesToLoad.Add("ou") | Out-Null

    $ous = $searcher.FindAll() | ForEach-Object {
        if ($_.Properties["ou"].Count -gt 0) { $_.Properties["ou"][0] }
    }

    foreach ($ou in ($ous | Sort-Object)) { $cbCity.Items.Add($ou) }
    if ($cbCity.Items.Count -gt 0) { $cbCity.SelectedIndex = 0 }
} catch {
    $cbCity.Items.AddRange(@('Paris','Lyon','Marseille','Bordeaux','Toulouse'))
    $cbCity.SelectedIndex = 0
}

# Fonctions Métier
function Remove-Accents([string]$s) {
    if (-not $s) { return "" }
    $d = $s.Normalize([Text.NormalizationForm]::FormD)
    ($d.ToCharArray() | Where-Object { [Globalization.CharUnicodeInfo]::GetUnicodeCategory($_) -ne 'NonSpacingMark' }) -join ''
}
function Refresh-Users {
    $lbUsers.Items.Clear()
    if (-not $cbCity.SelectedItem) { return }
    $ou = "OU=$($cbCity.SelectedItem),$baseOU"
    $t  = Remove-Accents $txtSearch.Text.Trim()
    $f  = if ($t) {"(&(objectCategory=person)(objectClass=user)(|(name=*$t*)(sn=*$t*)))"} else {"*"}
    Get-ADUser -SearchBase $ou -LDAPFilter $f -Properties Name,SamAccountName |
        Sort-Object Name | ForEach-Object { $lbUsers.Items.Add("$($_.Name) [$($_.SamAccountName)]") }
}
function Show-Details {
    if (-not $lbUsers.SelectedItem) { return }
    $sam = $lbUsers.SelectedItem -replace '.*\[(.+?)\]$','$1'
    $u   = Get-ADUser $sam -Properties *
    $detail["Path"].Text    = $u.DistinguishedName
    $detail["Nom"].Text     = $u.Surname
    $detail["Prenom"].Text  = $u.GivenName
    $detail["Service"].Text = $u.Department
    $detail["Matricule"].Text = $u.EmployeeID
    $detail["Mail"].Text    = $u.EmailAddress
    $detail["Session"].Text = $u.SamAccountName
    $chkForce.Checked       = ($u.pwdLastSet -eq 0)
    $lbGrps.Items.Clear()
    Get-ADPrincipalGroupMembership $sam | Select-Object -Expand Name | Sort-Object |
        ForEach-Object { $lbGrps.Items.Add($_) }
}
function Reset-Password {
    if (-not $lbUsers.SelectedItem) { return }
    $sam = $lbUsers.SelectedItem -replace '.*\[(.+?)\]$','$1'
    $id  = Get-ADUser $sam -Properties EmployeeID
    $pwd = "Foundever@$($id.EmployeeID)$(Get-RandomCode)"
    Set-ADAccountPassword -Identity $sam -Reset -NewPassword (ConvertTo-SecureString $pwd -AsPlainText -Force)
    Set-ADUser -Identity $sam -ChangePasswordAtLogon:$chkForce.Checked
    [Windows.Forms.MessageBox]::Show("Nouveau MDP : $pwd","Info")
}
function Disable-Account {
    if (-not $lbUsers.SelectedItem) { return }
    $sam = $lbUsers.SelectedItem -replace '.*\[(.+?)\]$','$1'
    $dst = "OU=Disabled Users,$baseOU"
    Move-ADObject -Identity (Get-ADUser $sam).DistinguishedName -TargetPath $dst
    Get-ADPrincipalGroupMembership $sam |
        ForEach-Object { Remove-ADGroupMember -Identity $_.Name -Members $sam -Confirm:$false }
    [Windows.Forms.MessageBox]::Show("Compte désactivé","Info")
    Refresh-Users
}
function Move-Group {
    if (-not $lbUsers.SelectedItem -or -not $lbGrps.SelectedItem -or -not $cbAllGrps.SelectedItem) {
        [Windows.Forms.MessageBox]::Show("Sélectionnez utilisateur, groupe source et groupe cible.","Attention")
        return
    }
    $sam = $lbUsers.SelectedItem -replace '.*\[(.+?)\]$','$1'
    $src = $lbGrps.SelectedItem; $dst = $cbAllGrps.SelectedItem
    $r   = [Windows.Forms.MessageBox]::Show(
        "Déplacer $sam de '$src' vers '$dst' ?","Confirmation",
        [Windows.Forms.MessageBoxButtons]::YesNo,[Windows.Forms.MessageBoxIcon]::Question
    )
    if ($r -ne 'Yes') { return }
    Remove-ADGroupMember -Identity $src -Members $sam -Confirm:$false
    Add-ADGroupMember    -Identity $dst -Members $sam
    [Windows.Forms.MessageBox]::Show("Déplacement OK","Info")
    Show-Details
}
function Add-Group {
    if (-not $lbUsers.SelectedItem -or -not $cbAllGrps.SelectedItem) { return }
    $sam = $lbUsers.SelectedItem -replace '.*\[(.+?)\]$','$1'
    $g   = $cbAllGrps.SelectedItem
    Add-ADGroupMember -Identity $g -Members $sam
    Show-Details
}
function Remove-Group {
    if (-not $lbUsers.SelectedItem -or -not $lbGrps.SelectedItem) { return }
    $sam = $lbUsers.SelectedItem -replace '.*\[(.+?)\]$','$1'
    $g   = $lbGrps.SelectedItem
    Remove-ADGroupMember -Identity $g -Members $sam -Confirm:$false
    Show-Details
}
