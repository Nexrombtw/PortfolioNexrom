# Fonctions utilitaires pour génerer des chiffres aléatoire 
function Get-RandomCode {
    $length = Get-Random -Minimum 4 -Maximum 7
    -join ((0..9) | Get-Random -Count $length)
}

function Remove-Accents {
    param([string]$Text)
    if (-not $Text) { return "" }
    $normalized = $Text.Normalize([System.Text.NormalizationForm]::FormD)
    ($normalized.ToCharArray() | Where-Object { [System.Globalization.CharUnicodeInfo]::GetUnicodeCategory($_) -ne 'NonSpacingMark' }) -join ''
}

function Load-FavoriteCity {
    if (Test-Path $configFile) {
        try {
            $config = Get-Content $configFile -Raw | ConvertFrom-Json
            if ($config -and $config.PSObject.Properties.Name -contains 'FavoriteCity') {
                return $config.FavoriteCity
            }
        } catch { } 
    }
    return $null
}

function Save-FavoriteCity {
    param([string]$City)
    $config = @{ FavoriteCity = $City }
    try {
        $config | ConvertTo-Json -Depth 1 | Set-Content -Path $configFile -Force
    } catch { } 
}

# Masque/Affiche les champs d'information utilisateur
function Hide-UserFields {

    $chkPwdMustChange.Visible = $false; $lblOui.Visible = $false; $btnResetPwd.Visible = $false; $btnDisable.Visible = $false

}
function Show-UserFields {
    foreach ($tb in $textboxes.Values) { $tb.Visible = $true }
    $chkPwdMustChange.Visible = $true; $lblOui.Visible = $true; $btnResetPwd.Visible = $true; $btnDisable.Visible = $true

}



# Recherche les utilisateurs dans l'AD
function Search-Users {
    Hide-UserFields; $comboUsers.Items.Clear(); $comboUsers.Text = ""; $txtPath.Text = ""
    $search = Remove-Accents $txtRecherche.Text.Trim()
    if (-not $search) { return }
    try {
        $users = Get-ADUser -Filter * -Properties Name, Surname, GivenName, SamAccountName | Where-Object {
            ($_.Name -and (Remove-Accents $_.Name) -like "*$search*") -or ($_.Surname -and (Remove-Accents $_.Surname) -like "*$search*") -or
            ($_.GivenName -and (Remove-Accents $_.GivenName) -like "*$search*") -or ($_.SamAccountName -and (Remove-Accents $_.SamAccountName) -like "*$search*")
        } | Sort-Object Name
        foreach ($u in $users) { $comboUsers.Items.Add("$(Remove-Accents $u.Name) [$($u.SamAccountName)]") }
        if ($comboUsers.Items.Count -gt 0) { $comboUsers.SelectedIndex = 0 }
    } catch { Write-Warning "Erreur recherche AD: $($_.Exception.Message)" }
}

# Remplit les champs avec les informations de l'utilisateur sélectionné
function Fill-UserInfo {
    Hide-UserFields
    if (-not $comboUsers.SelectedItem) { return }
    $sam = $comboUsers.SelectedItem.ToString() -replace '.*\[(.+?)\]$','$1'
    try {
        $u = Get-ADUser $sam -Properties * # Pourrait être optimisé en listant les propriétés
        if (-not $u) { [System.Windows.Forms.MessageBox]::Show("Utilisateur '$sam' non trouvé.", "Erreur"); return }
        
        $txtPath.Text = $u.DistinguishedName
        $textboxes["Nom"].Text = Remove-Accents $u.Surname; $textboxes["Prenom"].Text = Remove-Accents $u.GivenName
        $textboxes["Service"].Text = Remove-Accents $u.Department; $textboxes["Matricule"].Text = Remove-Accents $u.EmployeeID
        $textboxes["Description"].Text = Remove-Accents $u.Description; $textboxes["Adresse mail"].Text = Remove-Accents $u.EmailAddress
        $textboxes["Nom de session"].Text = Remove-Accents $u.SamAccountName
        $textboxes["Etat du compte"].Text = if ($u.Enabled) { "compte actif" } else { "compte desactive" }
        $textboxes["Date expiration"].Text = if ($u.AccountExpirationDate) { $u.AccountExpirationDate.ToString("dd/MM/yyyy HH:mm:ss") } else { "non verouillee" }
        $textboxes["Password expire"].Text = if ($u.PasswordNeverExpires) { "jamais" } else { "Expire" }
        $chkPwdMustChange.Checked = $u.PasswordExpired

        $listGroups.Items.Clear()
        $groups = Get-ADPrincipalGroupMembership -Identity $sam | Select-Object -ExpandProperty Name | Sort-Object
        foreach ($g in $groups) { $listGroups.Items.Add((Remove-Accents $g)) }
        Show-UserFields
    } catch { [System.Windows.Forms.MessageBox]::Show("Erreur chargement infos '$sam':`n$($_.Exception.Message)", "Erreur Critique"); return }
}

$btnResetPwd.Add_Click({
    if (-not $comboUsers.SelectedItem) { [System.Windows.Forms.MessageBox]::Show("Veuillez selectionner un agent.", "Selection requise"); return }
    $sam = $comboUsers.SelectedItem.ToString() -replace '.*\[(.+?)\]$','$1'
    try {
        $adUser = Get-ADUser $sam -Properties EmployeeID, Enabled
        if (-not $adUser) { [System.Windows.Forms.MessageBox]::Show("Utilisateur '$sam' non trouve.", "Erreur utilisateur"); return }
        if (-not $adUser.EmployeeID) { [System.Windows.Forms.MessageBox]::Show("Matricule introuvable pour $sam.", "Erreur matricule"); return }
        $newPwd = "Foundever@$($adUser.EmployeeID)$(Get-RandomCode)" # Assurer la conformité avec la politique de MDP
        if (-not $adUser.Enabled) { Set-ADUser -Identity $sam -Enabled $true }
        Set-ADAccountPassword -Identity $sam -Reset -NewPassword (ConvertTo-SecureString $newPwd -AsPlainText -Force)
        Set-ADUser -Identity $sam -ChangePasswordAtLogon $true
        $msg = "Mot de passe reinitialise pour $sam.`n`nNouveau mot de passe : $newPwd`n`nL'utilisateur devra changer ce MDP a sa prochaine connexion.`n"
        [System.Windows.Forms.Clipboard]::SetText($msg); [System.Windows.Forms.MessageBox]::Show($msg, "Reinitialisation reussie")
        Fill-UserInfo
    } catch { [System.Windows.Forms.MessageBox]::Show("Erreur reinitialisation MDP pour $sam :`n$($_.Exception.Message)", "Erreur critique") }
})

$btnDisable.Add_Click({
    if (-not $comboUsers.SelectedItem) { return }
    $sam = $comboUsers.SelectedItem.ToString() -replace '.*\[(.+?)\]$','$1'
    try {
        Disable-ADAccount -Identity $sam
        [System.Windows.Forms.MessageBox]::Show("Compte $sam desactive."); Fill-UserInfo
    } catch { [System.Windows.Forms.MessageBox]::Show("Erreur désactivation compte $sam :`n$($_.Exception.Message)") }
})
