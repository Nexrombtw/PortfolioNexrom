# Fonctions de stockage favori
function Load-Favorite {
    $favFile = Join-Path $env:APPDATA "UserFavorite.json"
    if (Test-Path $favFile) {
        try {
            return Get-Content $favFile -Raw | ConvertFrom-Json
        } catch {
            return @{}
        }
    }
    return @{}
}

function Save-Favorite($fav) {
    $favFile = Join-Path $env:APPDATA "UserFavorite.json"
    $fav | ConvertTo-Json -Depth 1 | Set-Content $favFile
}

# --- Fonction pour gérer la visibilité des détails utilisateur ---
function Set-UserDetailsVisibility($visible) {
    $txtPath.Visible = $visible
    foreach ($lbl in $detailLabels.Values) { $lbl.Visible = $visible }
    $chkForcePwd.Visible = $visible
    $btnResetPwd.Visible = $visible
    $btnDisable.Visible = $visible
    $btnMoveOU.Visible = $visible
    $lblGroups.Visible = $visible
    $lbGroups.Visible = $visible
    $lblAddGroups.Visible = $visible
    $cbAllGroups.Visible = $visible
    $btnMoveGroup.Visible = $visible
    $btnAddToGroup.Visible = $visible
    $btnRemoveFromGroup.Visible = $visible
    if (-not $visible) {
        $txtPath.Text = ""
        foreach ($lbl in $detailLabels.Values) { $lbl.Text = "-" }
        $lbGroups.Items.Clear()
    }
}

# --- Événements ---

# Rechercher utilisateurs
$null = $btnOK.Add_Click({
    $searchText = $txtRech.Text.Trim()
    $lbUsers.Items.Clear()
    if ([string]::IsNullOrWhiteSpace($searchText)) {
        [System.Windows.Forms.MessageBox]::Show("Entrez un terme de recherche.", "Info")
        return
    }
    try {
        $searchBase = "OU=Users,$($baseOUGlobal)"
        $filter = "(&(objectClass=user)(|"
        $filter += "(cn=*$searchText*)"
        $filter += "(givenName=*$searchText*)"
        $filter += "(sn=*$searchText*)"
        $filter += "(samAccountName=*$searchText*)"
        $filter += "))"
        $results = Get-ADUser -SearchBase $searchBase -LDAPFilter $filter -Properties * -ResultSetSize 1000
        foreach ($user in $results) {
            $lbUsers.Items.Add($user.SamAccountName)
        }
    } catch {
        [System.Windows.Forms.MessageBox]::Show("Erreur lors de la recherche : $($_.Exception.Message)", "Erreur")
    }
    # Cacher détails
    Set-UserDetailsVisibility $false
    $txtPath.Text = ""
})

# Lors de la sélection utilisateur
$null = $lbUsers.Add_SelectedIndexChanged({
    if ($lbUsers.SelectedItem) {
        $sam = $lbUsers.SelectedItem
        try {
            $user = Get-ADUser -Identity $sam -Properties *
            foreach ($field in $fields) {
                $value = $user.$($field.Key)
                if ($null -eq $value) { $value = "-" }
                $detailLabels[$field.Key].Text = "$value"
            }
            $dn = $user.DistinguishedName
            $txtPath.Text = "LDAP://$dn"
            Set-UserDetailsVisibility $true
        } catch {
            [System.Windows.Forms.MessageBox]::Show("Erreur : $($_.Exception.Message)", "Erreur")
        }
    }
})

# Double clic pour déselectionner
$null = $lbUsers.Add_DoubleClick({
    $lbUsers.ClearSelected()
    Set-UserDetailsVisibility $false
    $txtPath.Text = ""
})

# Favori

        $favName = $lbUsers.SelectedItem
        $favName | ConvertTo-Json | Set-Content (Join-Path $env:APPDATA "UserFavorite.json")
        [System.Windows.Forms.MessageBox]::Show("Favori sauvegardé : $favName", "Info")
    }
})

# Charger le favori si existant
if ($favoriteUser -and $favoriteUser -ne "") {
    try {
        $userFav = Get-ADUser -Identity $favoriteUser -Properties *
        foreach ($field in $fields) {
            $value = $userFav.$($field.Key)
            if ($null -eq $value) { $value = "-" }
            $detailLabels[$field.Key].Text = "$value"
        }
        $dn = $userFav.DistinguishedName
        $txtPath.Text = "LDAP://$dn"
        if ($lbUsers.Items.Contains($favoriteUser)) {
            $null = $lbUsers.SetSelected($lbUsers.Items.IndexOf($favoriteUser), $true)
        } else {
            $lbUsers.Items.Insert(0, $favoriteUser)
            $null = $lbUsers.SetSelected(0, $true)
        }
        Set-UserDetailsVisibility $true
    } catch {
        # ignore
    }
}

# Réinitialiser MDP
$null = $btnResetPwd.Add_Click({
    if (-not $lbUsers.SelectedItem) {
        [System.Windows.Forms.MessageBox]::Show("Sélectionnez un utilisateur d'abord.", "Info")
        return
    }
    $confirm = [System.Windows.Forms.MessageBox]::Show("Voulez-vous vraiment réinitialiser le mot de passe ?", "Confirmation", [System.Windows.Forms.MessageBoxButtons]::YesNo)
    if ($confirm -eq [System.Windows.Forms.DialogResult]::Yes) {
        $sam = $lbUsers.SelectedItem
        try {
            $user = Get-ADUser -Identity $sam -Properties EmployeeID, Enabled
            if (-not $user) {
                [System.Windows.Forms.MessageBox]::Show("Utilisateur introuvable.", "Erreur")
                return
            }
            if (-not $user.EmployeeID) {
                [System.Windows.Forms.MessageBox]::Show("Matricule introuvable.", "Erreur")
                return
            }
            $newPwd = "Foundever@$($user.EmployeeID)$(Get-RandomCode)"
            if (-not $user.Enabled) { Set-ADUser -Identity $sam -Enabled $true }
            Set-ADAccountPassword -Identity $sam -Reset -NewPassword (ConvertTo-SecureString $newPwd -AsPlainText -Force)
            Set-ADUser -Identity $sam -ChangePasswordAtLogon $true
            $msg = "Mot de passe réinitialisé pour $sam.`nNouveau mot de passe : $newPwd`nL'utilisateur devra changer son mot à sa prochaine connexion."
            [System.Windows.Forms.Clipboard]::SetText($msg)
            [System.Windows.Forms.MessageBox]::Show($msg, "Réinitialisation")
            # Recharger infos
            $user = Get-ADUser -Identity $sam -Properties *
            foreach ($field in $fields) {
                $value = $user.$($field.Key)
                if ($null -eq $value) { $value = "-" }
                $detailLabels[$field.Key].Text = "$value"
            }
            $dn = $user.DistinguishedName
            $txtPath.Text = "LDAP://$dn"
        } catch {
            [System.Windows.Forms.MessageBox]::Show("Erreur lors de la réinitialisation : $($_.Exception.Message)", "Erreur")
        }
    }
})

# Désactiver le compte
$null = $btnDisable.Add_Click({
    if (-not $lbUsers.SelectedItem) { return }
    $sam = $lbUsers.SelectedItem
    try {
        Disable-ADAccount -Identity $sam
        [System.Windows.Forms.MessageBox]::Show("Compte désactivé : $sam", "Info")
        $user = Get-ADUser -Identity $sam -Properties *
        foreach ($field in $fields) {
            $value = $user.$($field.Key)
            if ($null -eq $value) { $value = "-" }
            $detailLabels[$field.Key].Text = "$value"
        }
        $dn = $user.DistinguishedName
        $txtPath.Text = "LDAP://$dn"
    } catch {
        [System.Windows.Forms.MessageBox]::Show("Erreur : $($_.Exception.Message)", "Erreur")
    }
})
