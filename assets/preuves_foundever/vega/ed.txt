$configDir = Join-Path $env:APPDATA "InfoADScriptSettings_User"

# Fonctions utilitaires pour génerer des chiffres aléatoire
function Get-RandomCode {
    $length = Get-Random -Minimum 4 -Maximum 7
    -join ((0..9) | Get-Random -Count $length)
}

# Fonction pour normaliser le TERME DE RECHERCHE UTILISATEUR UNIQUEMENT
function Remove-Accents {
    param([string]$TextToNormalize)
    if (-not $TextToNormalize) { return "" }
    $normalized = $TextToNormalize.Normalize([System.Text.NormalizationForm]::FormD)
    ($normalized.ToCharArray() | Where-Object { [System.Globalization.CharUnicodeInfo]::GetUnicodeCategory($_) -ne 'NonSpacingMark' }) -join ''
}

function Load-FavoriteCity {
    if (Test-Path $configFile) {
        try {
            $config = Get-Content $configFile -Raw | ConvertFrom-Json
            if ($config -and $config.PSObject.Properties.Name -contains 'FavoriteCity') {
                return $config.FavoriteCity # Nom brut de la ville/OU (avec accents)
            }
        } catch { Write-Warning "Erreur lecture config ville fav: $($_.Exception.Message)" }
    }
    return $null
}

function Save-FavoriteCity {
    param([string]$City) # Nom brut de la ville/OU (avec accents)
    $config = @{ FavoriteCity = $City }
    try {
        $config | ConvertTo-Json -Depth 1 | Set-Content -Path $configFile -Force
    } catch { Write-Warning "Erreur sauvegarde config ville fav: $($_.Exception.Message)" }
}

#region Logique Applicative - Fonctions AD
function Search-Users {
    Hide-UserFields; $comboUsers.Items.Clear(); $comboUsers.Text = "" # $lblPathData.Text est déjà effacé par Hide-UserFields

    $searchTermFromInput = $txtRecherche.Text.Trim()
    $searchTermNormalized = Remove-Accents $searchTermFromInput

    if (-not $searchTermFromInput -and -not $comboVille.SelectedItem) {
        # Optionnel: retourner si aucun critère de recherche
        # return
    }

    try {
        $effectiveSearchBase = $ouBase

        if ($comboVille.SelectedItem) {
            $selectedCityName = $comboVille.SelectedItem.ToString()
            $effectiveSearchBase = "OU=$selectedCityName,$ouBase"

            if (-not (Get-ADOrganizationalUnit -Identity $effectiveSearchBase -ErrorAction SilentlyContinue)) {
                Write-Warning "L'OU '$effectiveSearchBase' pour la ville '$selectedCityName' n'a pas été trouvée. Recherche dans '$ouBase'."
                [System.Windows.Forms.MessageBox]::Show("L'OU pour la ville '$selectedCityName' est introuvable. La recherche sera effectuée dans la base parente des villes.", "Avertissement Ville")
                $effectiveSearchBase = $ouBase
            }
        }

        $filterParts = New-Object System.Collections.Generic.List[string]
        if ($searchTermNormalized) {
            $filterParts.Add("(|(name=*$searchTermNormalized*)(givenName=*$searchTermNormalized*)(sn=*$searchTermNormalized*)(samaccountname=*$searchTermNormalized*))")
        }

        $finalFilter = "*"
        if ($filterParts.Count -gt 0) {
            $finalFilter = "(&" + ($filterParts -join '') + ")"
        }

        $users = Get-ADUser -SearchBase $effectiveSearchBase -LDAPFilter $finalFilter -Properties Name, Surname, GivenName, SamAccountName | Sort-Object Name

        $comboUsers.Items.Clear()
        foreach ($u in $users) {
            $comboUsers.Items.Add("$($u.Name) [$($u.SamAccountName)]")
        }
        if ($comboUsers.Items.Count -gt 0) {
            $comboUsers.SelectedIndex = 0 # Ceci déclenchera Fill-UserInfo
        } else {
            [System.Windows.Forms.MessageBox]::Show("Aucun utilisateur trouvé pour les critères spécifiés dans '$effectiveSearchBase'.", "Recherche")
            Hide-UserFields # S'assurer de cacher les boutons si aucun user
        }
    } catch {
        Write-Warning "Erreur recherche AD: $($_.Exception.Message)"
        [System.Windows.Forms.MessageBox]::Show("Erreur lors de la recherche AD :`n$($_.Exception.Message)", "Erreur Recherche AD")
        Hide-UserFields
    }
}

function Fill-UserInfo {
    # Hide-UserFields est appelé au début de Search-Users, et les labels de données sont toujours visibles.
    # On efface juste le contenu avant de remplir.
    foreach ($key in $dataLabels.Keys) { $dataLabels[$key].Text = "-" }
    $lblPathData.Text = "" # Effacer le path précédent

    if (-not $comboUsers.SelectedItem) { Hide-UserFields; return } # S'il n'y a plus de sélection, cacher les boutons

    $selectedUserString = $comboUsers.SelectedItem.ToString()
    $sam = $selectedUserString -replace '.*\[(.+?)\]$','$1'
    try {
        $propertiesToLoad = @("Name", "Surname", "GivenName", "SamAccountName", "Department", "EmployeeID",
                              "Description", "EmailAddress", "Enabled", "AccountExpirationDate", "physicalDeliveryOfficeName",
                               "PasswordNeverExpires", "DistinguishedName", "pwdLastSet" )
        $u = Get-ADUser $sam -Properties $propertiesToLoad

        if (-not $u) { [System.Windows.Forms.MessageBox]::Show("Utilisateur '$sam' non trouvé.", "Erreur"); Hide-UserFields; return }

        $lblPathData.Text = $u.DistinguishedName # Remplir le label Path
        $dataLabels["Nom"].Text = $u.Surname
        $dataLabels["Prenom"].Text = $u.GivenName
        $dataLabels["Service"].Text = $u.Department
        $dataLabels["Matricule"].Text = $u.EmployeeID
        $dataLabels["Description"].Text = $u.Description
        $dataLabels["Adresse mail"].Text = $u.EmailAddress
        $dataLabels["Nom de session"].Text = $u.SamAccountName
        $dataLabels["Etat du compte"].Text = if ($u.Enabled) { "compte actif" } else { "compte desactive" }
        $dataLabels["Date expiration"].Text = if ($u.AccountExpirationDate) { $u.AccountExpirationDate.ToString("dd/MM/yyyy HH:mm:ss") } else { "non verouillee" }
        $dataLabels["Password expire"].Text = if ($u.PasswordNeverExpires) { "jamais" } else { "Expire (selon politique)" }
        $dataLabels["Ville utilisateur"].Text = $u.physicalDeliveryOfficeName # Remplir le label Ville utilisateur

        if ($null -ne $u.pwdLastSet -and $u.pwdLastSet -eq 0) {
            $lblPwdMustChange.Tag = $true
            $lblPwdMustChange.Text = "[X] L'utilisateur doit changer son mot de passe"
            $lblPwdMustChange.ForeColor = [System.Drawing.Color]::OrangeRed
        } else {
            $lblPwdMustChange.Tag = $false
            $lblPwdMustChange.Text = "[  ] L'utilisateur doit changer son mot de passe"
            $lblPwdMustChange.ForeColor = $script:theme.Text
        }

        $listGroups.Items.Clear()
        $groups = Get-ADPrincipalGroupMembership -Identity $sam | Select-Object -ExpandProperty Name | Sort-Object
        foreach ($g in $groups) {
            $listGroups.Items.Add($g)
        }
        Show-UserFields # Rendre visibles les boutons d'action et la liste des groupes
    } catch {
        $errorMessage = "Erreur chargement infos pour '$sam':`n$($_.Exception.Message)"
        if ($_.Exception.InnerException) { $errorMessage += "`nInner Exception: $($_.Exception.InnerException.Message)" }
        [System.Windows.Forms.MessageBox]::Show($errorMessage, "Erreur Critique")
        Hide-UserFields
        return
    }
}

$btnResetPwd.Add_Click({
    if (-not $comboUsers.SelectedItem) { [System.Windows.Forms.MessageBox]::Show("Veuillez selectionner un agent.", "Selection requise"); return }
    $sam = ($comboUsers.SelectedItem.ToString() -replace '.*\[(.+?)\]$','$1')
    try {
        $adUser = Get-ADUser $sam -Properties EmployeeID, Enabled
        if (-not $adUser) { [System.Windows.Forms.MessageBox]::Show("Utilisateur '$sam' non trouve.", "Erreur utilisateur"); return }
        if (-not $adUser.EmployeeID) { [System.Windows.Forms.MessageBox]::Show("Matricule introuvable pour $sam.", "Erreur matricule"); return }
        $newPwd = "Foundever@$($adUser.EmployeeID)$(Get-RandomCode)"
        if (-not $adUser.Enabled) { Set-ADUser -Identity $sam -Enabled $true }
        Set-ADAccountPassword -Identity $sam -Reset -NewPassword (ConvertTo-SecureString $newPwd -AsPlainText -Force)

        Set-ADUser -Identity $sam -ChangePasswordAtLogon:([bool]$lblPwdMustChange.Tag)

        $userMessage = "Mot de passe réinitialisé pour $sam.`nNouveau mot de passe : $newPwd"
        if ([bool]$lblPwdMustChange.Tag) { $userMessage += "`nL'utilisateur devra changer ce mot de passe à la prochaine connexion." }
        else { $userMessage += "`nL'utilisateur NE devra PAS changer ce mot de passe à la prochaine connexion." }
        $clipboardText = "Utilisateur: $sam`nNouveau mot de passe: $newPwd"
        [System.Windows.Forms.Clipboard]::SetText($clipboardText); [System.Windows.Forms.MessageBox]::Show($userMessage, "Réinitialisation réussie"); Fill-UserInfo
    } catch { [System.Windows.Forms.MessageBox]::Show("Erreur reinitialisation MDP pour $sam :`n$($_.Exception.Message)", "Erreur critique") }
})

$btnDisable.Add_Click({
    if (-not $comboUsers.SelectedItem) { return }
    $sam = ($comboUsers.SelectedItem.ToString() -replace '.*\[(.+?)\]$','$1')
    try {
        Disable-ADAccount -Identity $sam
        [System.Windows.Forms.MessageBox]::Show("Compte $sam desactive."); Fill-UserInfo
    } catch { [System.Windows.Forms.MessageBox]::Show("Erreur désactivation compte $sam :`n$($_.Exception.Message)") }
})
